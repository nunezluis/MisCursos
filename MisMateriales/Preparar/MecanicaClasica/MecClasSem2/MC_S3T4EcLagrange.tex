\section{Principio de Mínima Acción}
\frame{
  \frametitle{Principio de Mínima Acción}
   \begin{itemize}  
  	\item<1-> Consideremos un sistema descrito por $s$ coordenadas generalizadas $\left\{q_1, q_2, \ldots, q_s\right\}$ y sus $s$ velocidades generalizadas $\left\{\dot{q}_1, \dot{q}_2, \ldots, \dot{q}_s\right\}$. 
  	\item<2-> Definimos una función $\mathcal{L}$ de $\left\{q_j\right\},\left\{\dot{q}_j\right\}$ y $t$, de la forma $\mathcal{L}\left(q_j, \dot{q}_j, t\right) \equiv T-V, \quad j=1,2, \ldots, s$, donde $T$ y $V$ son la energía cinética y la energía potencial del sistema, respectivamente.
	\item<3-> El estado del sistema, en $t=t_1$ y $t=t_2$, está descrito por $t_1:\left\{q_j\left(t_1\right)\right\},\left\{\dot{q}_j\left(t_1\right)\right\}$ y  $t_2:\left\{q_j\left(t_2\right)\right\},\left\{\dot{q}_j\left(t_2\right)\right\}$
	\item<4-> El Principio de mínima acción, implica que la evolución del sistema entre el estado en $t_1$ al $t_2$ es tal que el valor de la integral definida
$S=\int_{t_1}^{t_2} \mathcal{L}\left(q_j, \dot{q}_j, t\right) {\rm d} t,$
denominada la acción del sistema, sea mínima; es decir, $\delta S=0$ ( $S$ es un extremo).
    \end{itemize}
}
