%%%%% Diapo 2
\subsection{El funcional y la Ecuación Euler}
\frame{
  \frametitle{El funcional y la ecuación de Euler}
   \begin{itemize}  
  	\item<1->  Fijamos el punto $(x_1, y_1) = (0,0)$ y la dirección del eje $y$ hacia abajo.
	\item<2->  Si $v$ es la magnitud de la velocidad en un punto de la trayectoria, entonces el elemento de tiempo para recorrer una distancia infinitesimal ${\rm d}s$ a lo largo de la trayectoria es $d t=\frac{{\rm d} s}{v}$.
	\item<3-> El tiempo total para ir del punto 1 al punto 2 es $t_{1 \rightarrow 2}=\int_1^2 \frac{{\rm d} s}{v}=\int_1^2 \frac{\sqrt{{\rm d} x^2 +{\rm d} y^2}}{v} =\int_{y_1}^{y_2} \sqrt{\frac{1+\left(x^{\prime}\right)^2}{2 g y}} {\rm d} y $
	\item<4-> Identificamos el función del funcional $f\left(x, x^{\prime}, y\right)=\sqrt{\frac{1+\left(x^{\prime}\right)^2}{2 g y}}$
	\item<5-> La ecuación de Euler correspondiente es $\frac{{\rm d}}{{\rm d} y}\left(\frac{\partial f}{\partial x^{\prime}}\right)-\frac{\partial f}{\partial x}= 0 \Rightarrow \frac{x^{\prime}}{\sqrt{2 g y} \sqrt{1+\left(x^{\prime}\right)^2}}=c=$ constante
	\item<6-> En este caso la ecuación de Euler para  $f\left(x, x^{\prime}, y\right)$ resulta más sencilla que para $f\left(y, y^{\prime}, x\right)$, porque $\frac{\partial f}{\partial x}=0$
	\item<7-> Entonces $x^{\prime}  =\frac{{\rm d} x}{{\rm d} y}=\sqrt{\frac{2 g c^2 y}{1-2 g c^2 y}} \Rightarrow x  =\int \sqrt{\frac{y}{\frac{1}{2 g c^2}-y}} {\rm d} y=\int \sqrt{\frac{y}{2 R-y}} {\rm d} y, $ con $2 R \equiv 1 / 2 g c^2$
    \end{itemize}
}

%%%%% Diapo 2
\subsection{La trayectoria}
\frame{
  \frametitle{La trayectoria}
  \begin{itemize}  
  	\item<1-> Con un cambio de variable $y=R(1-\cos \theta), \quad {\rm d} y=R {\rm sen}\, \theta {\rm d} \theta$, tendremos $x  =R \int \sqrt{\frac{(1-\cos \theta)}{(1+\cos \theta)}} {\rm sen}\, \theta \;  {\rm d} \theta = R \int(1-\cos \theta) {\rm d} \theta $
	\item<2-> La trayectoria queda parametrizada por \\ $y=R(1-\cos \theta)$ y $x=R(\theta-{\rm sen}\, \theta)$. \\ La ecuación de la cicloide que pasa por $\left(x_1, y_1\right)=(0,0)$, con $k=0$.
	\item<3-> La constante $R$ se determina con el punto $\left(x_2, y_2\right)$ y da al valor del radio de la circunferencia que genera la cicloide.
	\item<4-> La trayectoria de tiempo mínimo es un arco de cicloide que pasa por los puntos dados $\theta_{\frac{\pi}{2}}  \Rightarrow y=R, x=\frac{\pi}{2} R$, $\theta_{\pi} \Rightarrow x=\pi R, y=2 R$ y $\theta_{2 \pi} \Rightarrow x=2 \pi R, y=0$
  	\begin{figure}[t]
		\includegraphics[width=2.4in]{Figuras/Cicloide.png} 
	\end{figure}  

   \end{itemize}
}

\section{Principio de Fermat}
\frame{
  \frametitle{Principio de Fermat}
  \begin{itemize}  
  	\item<1-> La luz se propaga entre dos puntos siguiendo la trayectoria que corresponde al tiempo mínimo
	\item<2-> La velocidad de la luz en un medio con índice de refracción $n$ es $v=c / n$, donde $c$ es la velocidad de la luz en el vacío.
	  	\begin{figure}[t]
		\includegraphics[width=1.4in]{Figuras/Fermat.png} 
	\end{figure}
	\item<3-> El tiempo para viajar entre los puntos 1 y 2 es 
	$t_{1 \rightarrow 2} = \int_1^2 \frac{{\rm d} s}{v}=\int_1^2 n \frac{\sqrt{{\rm d} x^2+{\rm d} y^2}}{c} =\frac{1}{c} \int_{y_1}^{y_2} n \sqrt{1+\left(x^{\prime}\right)^2} {\rm d} y$
	\item<4-> Tenemos dos índices de refracción $n=n_1$, para $y>0$, y $n=n_2$, para $y<0$, tal que $\frac{d n}{d y} \neq 0$ sólo en $y=0$.  
   \end{itemize}
}
%%%%% Diapo 2
\frame{
  \frametitle{Principio de Fermat}
  \begin{itemize}  
  	\item<1-> Entonces podemos identificar $f\left(x, x^{\prime}, y\right)=n \sqrt{1+\left(x^{\prime}\right)^2}$, que satisface la ecuación de Euler, $\frac{d}{d y}\left(\frac{\partial f}{\partial x^{\prime}}\right)-\frac{\partial f}{\partial x}=0$
	\item<2-> Con lo cual $\frac{n x^{\prime}}{\sqrt{1+x^{\prime 2}}}=$ cte. 
	\item<3-> Adicionalmente, $x^{\prime}=\frac{d x}{d y}=-\tan \theta$, con $\theta$ el ángulo de la trayectoria de la luz con el eje $y$ (normal a la interfase). 
	\item<4-> Con lo cual obtenemos $-\frac{n \tan \theta}{\sqrt{1+\tan ^2 \theta}} =\text { cte. } \Rightarrow n \operatorname{sen} \theta  =\text { cte. }$, 
	\item<5-> Finalmente, implica la ley de refracción, $n_1 \operatorname{sen} \theta_1=n_2 \operatorname{sen} \theta_2$
   \end{itemize}
   }

\section{Recapitulando}
\frame{
  \frametitle{Recapitulando}
  \begin{itemize}  
  	\item<1-> La distancia más corta no siempre es la más rápida; {\it los principios variacionales} determinan los caminos óptimos.
	\item<2-> La integral del tiempo es $ t_{1 \to 2} = \int_{y_1}^{y_2} \frac{\sqrt{1+(x')^2}}{\sqrt{2gy}} \, dy$.
	\item<3-> El problema se resuelve mediante la {\it ecuación de Euler-Lagrange}, genera una primera integral que relaciona la pendiente de la trayectoria con la energía.
	\item<4-> La trayectoria resultante es un {\it segmento de una cicloide}, parametrizada como:   $ x = R(\theta - \operatorname{sen}\,\theta), \quad y = R(1 - \cos \theta )$, donde $R$ se determina por las condiciones de frontera.
	\item<5-> El {\it principio de Fermat} establece que la luz sigue el {\it camino del menor tiempo}, lo que conduce a la {\it ley de Snell} de la refracción: $n_1 \operatorname{sen}\,\theta_1 = n_2 \operatorname{sen}\,\theta_2$
   \end{itemize}
}
