\section{Las transformaciones canónicas $\left\{p_i, q_i\right\} \rightarrow\left\{P_i, Q_i\right\}=\left\{\alpha_i, \beta_i\right\}$}
\frame{
  \frametitle{Las transformaciones canónicas}
   \begin{itemize}  
  	\item<1-> Una transformación canónica $Q_i=Q_i\left(q_j, p_j, t\right), P_i=P_i\left(q_j, p_j, t\right)$, permite encontrar las soluciones de las ecuaciones de Hamilton.
	\item<2-> Esa transformación $Q_i=Q_i\left(q_j, p_j, t\right), P_i=P_i\left(q_j, p_j, t\right)$ lleva $\mathcal{H}\left(q_j, p_j, t\right)\rightarrow \mathcal{H}^{\prime}\left(Q_i, P_i, t\right)$ un hamiltoniano en el cual una (o varias) coordenada $Q_j$ o/y $P_j$ son cíclicas
	\item<3-> Supongamos una transformación canónica $\left\{q_i, p_i, t\right\} \rightarrow \left\{P_i, Q_i, t\right\}$ tal que, $\left(P_i, Q_i\right)$, las $2 s$ nuevas coordenadas y momentos  son constantes.
	\item<4-> Esas $2 s$ constantes $Q_i$ y $P_i$ pueden expresarse en función de las $2 s$ condiciones iniciales:  $Q_i=q\left(q_j(0), p_j(0)\right), P_i=p\left(q_j(0), p_j(0)\right)$.
	\item<5-> La transformación canónica que relacionan las nuevas y viejas variables proporcionan directamente la solución del problema del movimiento, $q_i=q\left(q_j(0), p_j(0), t\right), \quad p_i=p\left(q_j(0), p_j(0), t\right)$.
	\item<6-> Si  la transformación canónica conduce a nuevos momentos y coordenadas constantes, $P_i \equiv \alpha_i=$ cte, $Q_i \equiv \beta_i=$ cte, tal que $\mathcal{H}^{\prime}\left(Q_i, P_i\right)=0$, entonces existe una función generadora $\mathcal{F}$ tal que $\frac{\partial F}{\partial t}+\mathcal{H}=0$
    \end{itemize}
}

\frame{
\frametitle{El Principio de Mínima Acción}
\begin{itemize}  
	\item<1-> Consideremos la acción $S=\int_{t_1}^{t_2} \mathcal{L}\left(q_i, \dot{q}_i, t\right) d t $

	\item<5-> La acción dependerá de las trayectorias y del tiempo, $S=S\left(q_i, t\right)$.

\section{La ecuación de Hamilton-Jacobi}
\frame{
\frametitle{La ecuación de Hamilton-Jacobi}
\begin{itemize}  
	\item<1-> La derivada temporal de la acción es $\frac{d S}{d t}=\sum_{i=1}^s \frac{\partial S}{\partial q_i} \dot{q}_i+\frac{\partial S}{\partial t}$.	
	\item<2-> Por otro lado, si $t_2=t$ (variable), la definición de la acción implica que $\frac{d S}{d t}=\mathcal{L}=\sum_{i=1}^s p_i \dot{q}_i-\mathcal{H}\left(p_i, q_i, t\right)$.
	\item<3-> Comparando obtenemos $ p_i=\frac{\partial S}{\partial q_i}\left(q_i, t\right)$ y $\frac{\partial S}{\partial t}\left(q_i, t\right)+\mathcal{H}\left(p_i, q_i, t\right)=0$, 
	\item<4-> Las cuales se pueden expresar como $\frac{\partial S}{\partial t}\left(q_i, t\right)+\mathcal{H}\left(\frac{\partial S}{\partial q_i}, q_i, t\right)=0$. Que es la Ecuación de Hamilton Jacobi. 
	\item<5-> La acción $S$ puede interpretarse como una función generadora capaz de producir la transformación canónica.
	\item<6-> Más aún, la acción $S$ puede interpretarse como una función generadora tipo $\mathcal{F}_2\left(q_i, P_i, t\right)$, tal que $P_i=\alpha_i=$ cte, $Q_i=\beta_i=$ cte.
	\item<7->La derivada total $\frac{d \mathcal{F}_2}{d t}=\sum_{i=1}^s\left(\frac{\partial \mathcal{F}_2}{\partial q_i} \dot{q}_i+\frac{\partial \mathcal{F}_2}{\partial P_i} \dot{P}_i\right)+\frac{\partial \mathcal{F}_2}{\partial t}$ $\Rightarrow \frac{d \mathcal{F}_2}{d t}=\sum_{i=1}^s p_i \dot{q}_i-\mathcal{H}$
	\item<7-> Donde hemos usado: $p_i=\frac{\partial \mathcal{F}_2}{\partial q_i}$ y $\dot{P}_i=0$.
	
\end{itemize}
}

\frametitle{Título transparencia}
\begin{itemize}  
	\item<1-> Comparando las relaciones de $\mathcal{F}_2\left(q_i, P_i, t\right)$ y $S\left(q_i, P_i, t\right)$, tenemos 
	$\begin{array}{l|l}
	p_i=\frac{\partial \mathcal{F}_2}{\partial q_i}\left(q_i, P_i, t\right) & p_i=\frac{\partial S}{\partial q_i}\left(q_i, P_i, t\right)=p_i\left(q_i, P_i, t\right) \\
	Q_i=\frac{\partial \mathcal{F}_2}{\partial P_i}\left(q_i, P_i, t\right) & Q_i=\frac{\partial S}{\partial P_i}\left(q_i, P_i, t\right)=Q_i\left(q_i, P_i, t\right) \\
	\mathcal{H}+\frac{\partial \mathcal{F}_2}{\partial t}=\mathcal{H}^{\prime} & \mathcal{H}+\frac{\partial S}{\partial t}=0
	\end{array}$ \\ donde $P_i=$ cte $=\alpha_i$ y $Q_i=$ cte $=\beta_i$.
	\item<2-> Si $\mathcal{H}^{\prime}\left(P_i, Q_i\right)=$ cte y existe una transformación canónica $\left\{p_i, q_i\right\} \rightarrow\left\{P_i, Q_i\right\}=\left\{\alpha_i, \beta_i\right\}$, generada por $\mathcal{F}_2=S$, tal que $\mathcal{H}^{\prime}\left(P_i, Q_i\right) \equiv \mathcal{H}^{\prime}\left(\alpha_i, \beta_i\right)=0$, entonces se satisface ecuación de Hamilton-Jacobi, $\frac{\partial S}{\partial t}+\mathcal{H}=0$
	\item<3-> La solución $S\left(q_i, P_i, t\right)$ de la ecuación de Hamilton-Jacobi genera la transformación $\left\{p_i, q_i, t\right\} \rightarrow\left\{P_i, Q_i, t\right\}$.
	\item<4->  Las constantes $P_i ,Q_i \leftrightarrow \alpha_i, \beta_i$ se expresan en términos de las $2 s$ condiciones iniciales $\left(q_i(0), p_i(0)\right)$.
	\item<5-> La solución de la ecuación de Hamilton-Jacobi para un sistema provee la trayectoria $q_i(t)=q_i\left(q_i(0), p_i(0), t\right)$ y $p_i(t)=p_i\left(q_i(0), p_i(0), t\right)$.

\end{itemize}
}

\section{Sección}
\frame{
\frametitle{Título transparencia}
\begin{itemize}  
	\item<1-> Matemáticamente, la ecuación de Hamilton-Jabobi es una ecuación en derivadas parciales de primer orden para $S\left(q_i, t\right)$ con $s+1$ variables,
	$\frac{\partial S}{\partial t}\left(q_1, \ldots, q_s, t\right)+H\left(q_1, \ldots, q_s, \frac{\partial S}{\partial q_1}, \frac{\partial S}{\partial q_2} \ldots, \frac{\partial S}{\partial q_s}, t\right)=0$
	\item<2-> La función $S$ no es incógnita, sólo aparecen sus derivadas.
	\item<3-> Si $S$ es solución, entonces $\tilde{S} =S+C_1$ también lo es. 
	\item<4-> Una de las $(s+1)$ constantes de integración es irrelevante.
	\item<5-> Para que la acción tenga la forma $S\left(q_i, P_i, t\right)$, tomamos las $s$ constantes $P_i=\alpha_i$. 
	\item<6-> La solución $ S=S\left(q_1, \ldots, q_s, P_1, \ldots, P_s, t\right)=S\left(q_i, \alpha_i, t\right)$.
	\item<7-> Si el Hamiltoniano $\mathcal{H}$ es independiente del tiempo, entonces es una constante, igual a la energía total del sistema, $\mathcal{H}\left(q_i, p_i\right)= cte =\mathcal{E}$. 
	\item<8-> Suponemos que la solución $S$ tiene la forma 
	$S\left(q_i, P_i, t\right)=S\left(q_i, \alpha_i, t\right)=\mathcal{W}\left(q_i, P_i\right)- \mathcal{E} t=\mathcal{W}\left(q_i, \alpha_i\right)-\mathcal{E} t$. \\ Una de las $s$ constantes $\alpha_i$ es $\mathcal{E}$
	\item<9-> La función $\mathcal{W}\left(q_i, P_i\right)=\mathcal{W}\left(q_i, \alpha_i\right)$ se llama función característica o principal de Hamilton.
\end{itemize}
}

\section{Ejemplo: Una vez mas el Oscilador Armónico}
\frame{
\frametitle{Ejemplo: Una vez mas el Oscilador Armónico}
Ecuación de Hamilton-Jacobi para un oscilador armónico simple y la acción asociada a este sistema.
\begin{itemize}  
	\item<1-> El Hamiltoniano es $\mathcal{H}(q, p)=\frac{1}{2 m}\left(p^2+m^2 \omega^2 q^2\right)$
	\item<2-> La ecuación de Hamilton-Jacobi es $\frac{\partial S}{\partial t}(q, t)+\mathcal{H}(p, q)=0$
	\item<3-> Como $p=\frac{\partial S}{\partial q}(q, t)$, obtenemos $\frac{\partial S}{\partial t}+\frac{1}{2 m}\left[\left(\frac{\partial S}{\partial q}\right)^2+m^2 \omega^2 q^2\right]=0$.\\  Una ecuacion diferencial parcial de primer orden
	\item<5-> Puesto que $\frac{\partial \mathcal{H}}{\partial t}=0$, el Hamitoniano es constante e igual a la energía total del sistema, $\mathcal{H}=\mathcal{E}$
	\item<6-> Buscamos una solución por separación de variables, $S(q, \mathcal{E}, t)=W(q, \mathcal{E})-\mathcal{E} t$, con $P=E=\alpha$   constante de integración 
	\item<7-> Entonces, $\frac{1}{2 m}\left[\left(\frac{\partial \mathcal{W} }{\partial q}\right)^2+m^2 \omega^2 q^2\right]=E \Rightarrow \frac{\partial \mathcal{W} }{\partial q} =\left(2 m \mathcal{E}-m^2 \omega^2 q^2\right)^{1 / 2} \Rightarrow$,  
$\mathcal{W}(q, E)  =\int\left(2 m \mathcal{E}-m^2 \omega^2 q^2\right)^{1 / 2} d q \equiv S(q, \mathcal{E}, t) +Et$
\end{itemize}
}

\section{Comparando Métodos.}
\subsection{Comparando Métodos: Conceptos y Técnicas}
\frame{
\frametitle{Comparando Métodos: Conceptos y Técnicas}
\begin{itemize}  
	\item<1-> {\bf Los Conceptos de los formalismos}
	\begin{itemize}
  	\item \textbf{Lagrangiano:} Se basa en el principio de mínima acción \( \delta S = 0 \), con \( S = \int \mathcal{L} \, dt \). Utiliza coordenadas generalizadas \( q_i \) y velocidades \( \dot{q}_i \). Las ecuaciones de movimiento son las ecuaciones de Euler-Lagrange.
	%
  	\item \textbf{Hamiltoniano:} Surge de una transformación de Legendre del lagrangiano. Las variables son \( q_i \) y sus momentos conjugados \( p_i \). Las ecuaciones de movimiento son de primer orden.
  	\item \textbf{Hamilton-Jacobi:} Busca una función generadora \( S(q,t) \) que satisface una ecuación en derivadas parciales no lineal. Es una reformulación avanzada que unifica el análisis dinámico con métodos integrables.
	\end{itemize}
	\item<2-> {\bf Las Técnicas}
	\begin{itemize}
  	\item {\bf Lagrangiano}: espacio de configuración y ecuaciones diferenciales de segundo orden.
  	\item {\bf Hamiltoniano}: espacio de fases, y ecuaciones de primer orden.
  	\item {\bf Hamilton-Jacobi} reduce el problema a una única ecuación en derivadas parciales; las soluciones de \( S \) generan toda la dinámica.
  	\item {\bf Simetrías y  cantidades conservadas} se identifican más fácilmente en el formalismo Hamiltoniano, aunque el teorema de Noether también se aplica al lagrangiano.
	\end{itemize}
\end{itemize}
}

\subsection{Ventajas y Limitaciones}
\frame{
\frametitle{Ventajas y Limitaciones}
\begin{itemize}  
	\item<1-> {\bf Ventajas y Limitaciones}
	\begin{itemize}
  	\item \textbf{Lagrangiano:} Ideal para sistemas con ligaduras y simetrías. Se adapta bien a 	sistemas con coordenadas generalizadas. Menos eficiente en análisis de estabilidad.
  	\item \textbf{Hamiltoniano:} Proporciona una visión clara del espacio de fases y permite análisis cualitativos. Es la base para el paso a la mecánica cuántica.
  	\item \textbf{Hamilton-Jacobi:} Potente para sistemas integrables. Su conexión con la mecánica ondulatoria lo hace fundamental para el tránsito clásico-cuántico. Sin embargo, resolver la ecuación de Hamilton-Jacobi puede ser muy difícil.
	\end{itemize}
	\item<2-> {\bf Resumiento}
	\begin{itemize}
  	\item Los tres métodos describen la misma física, pero desde perspectivas distintas.
  	\item El enfoque lagrangiano es más geométrico y útil con ligaduras.
  	\item El hamiltoniano es estructuralmente más rico y se presta al análisis de conservación y estabilidad.
  	\item El enfoque de Hamilton-Jacobi es el más general y conecta elegantemente con la mecánica cuántica.
	\end{itemize}
\end{itemize}
}

\subsection{El Oscilador Armónico: Tres Enfoques}
\frame{
\frametitle{El Oscilador Armónico: Tres Enfoques 1/2}
Partícula de masa \( m \), sujeta a una fuerza $F = -kx \quad \Rightarrow \quad V(x) = \frac{1}{2}k x^2$
\begin{itemize}  
	\item<1-> {\bf Método Lagrangeano}
	\begin{itemize}
		\item Lagrangiano:  $\mathcal{L}(x, \dot{x}) = \frac{1}{2}m\dot{x}^2 - \frac{1}{2}k x^2$
		\item Ecuación de Euler-Lagrange $\frac{d}{dt} \left( \frac{\partial \mathcal{L}}{\partial \dot{x}} \right) = \frac{\partial \mathcal{L}}{\partial x} \quad \Rightarrow \quad m\ddot{x} + kx = 0$
		\item Solución: $x(t) = A \cos(\omega t) + B \sin(\omega t), \quad \omega = \sqrt{\frac{k}{m}}$
	\end{itemize}
	\item<2-> {\bf Ventajas del enfoque Lagrangiano}
	\begin{itemize}
  		\item Se basa en el principio de mínima acción.
  		\item Ideal para sistemas con ligaduras y coordenadas generalizadas.
  		\item Geométricamente intuitivo y extendible a campos.
	\end{itemize}
	\item<3-> {\bf Método Hamiltoniano}
	\begin{itemize}
		\item Hamiltoniano a partir de \( \mathcal{L} \) $\Rightarrow p = \frac{\partial \mathcal{L}}{\partial \dot{x}} = m \dot{x}, \quad \mathcal{H} = \frac{p^2}{2m} + \frac{1}{2}kx^2$
		\item Ecuaciones de Hamilton $\dot{x} = \frac{\partial \mathcal{H}}{\partial p} = \frac{p}{m}, \quad \dot{p} = -\frac{\partial \mathcal{H}}{\partial x} = -kx$
		\item \textbf{Resultado:} $\ddot{x} + \frac{k}{m}x = 0$ 
	\end{itemize}
	\item<4-> {\bf Ventajas del enfoque Hamiltoniano}
\begin{itemize}
  \item Describe la evolución en el espacio de fases.
  \item Revela estructura conservativa y simetrías.
  \item Puente natural hacia mecánica cuántica y estadística.
\end{itemize}
\end{itemize}
}

  		\item \textbf{Hamiltoniano:} Estructural, simétrico, fundamental en física moderna.
  		\item \textbf{Hamilton-Jacobi:} Avanzado, integra el sistema completo, conecta con teoría cuántica.
	\end{itemize}
\end{itemize}

$$
S\left(q_i, P_i, t\right)=W\left(q_j, P_i\right)+P_k q_k-E t, \quad j \neq k
$$


Entonces,

$$
p_k=\frac{\partial S}{\partial q_k}=P_k=\alpha_k=\text { cte. }
$$
