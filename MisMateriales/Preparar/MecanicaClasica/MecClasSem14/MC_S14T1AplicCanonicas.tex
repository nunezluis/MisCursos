\section{Transformaciones canónicas infinitesimales}
\frame{
  \frametitle{Transformaciones canónicas infinitesimales}
   \begin{itemize}  
  	\item<1-> Consideremos una transformación infinitesimal de la forma $Q_i  =q_i+\epsilon f_i\left(q_j, p_j\right)$ y $P_i  =p_i+\epsilon g_i\left(q_j, p_j\right)$, \\ donde $f_i\left(q_j, p_j\right)$ y, $g_i\left(q_j, p_j\right)$ son funciones dadas y $\epsilon \lll 1$.
	\item<2-> Podemos considerar que la transformación $\{q_i, p_i \} \rightarrow \{Q_i, P_i \}$ es una desviación infinitesimal de una transformación identidad. 
	\item<3-> Para que la transformación sea canónica debe existir una función generadora $\mathcal{F}_2\left(q_i, P_i\right)=\sum_{i=1}^s q_i P_i+\epsilon \mathcal{G}\left(q_i, P_i\right)$, con $\mathcal{G}\left(q_i, P_i\right)$ una función a determinar.
	\item<4-> La función generadora es una pequeña desviación de la función generadora para la transformación identidad
	\item<5-> La transformación canónica generada por  $\mathcal{F}_2\left(q_i, P_i\right)$ es $p_i  =\frac{\partial \mathcal{F}_2}{\partial q_i}=P_i+\epsilon \frac{\partial \mathcal{G}}{\partial q_i} \quad$ y $\quad Q_i  =\frac{\partial \mathcal{F}_2}{\partial P_i}=q_i+\epsilon \frac{\partial \mathcal{G}}{\partial P_i}$
	\item<6-> Entonces $f_i\left(q_j, p_j\right)  =\frac{\partial \mathcal{G}}{\partial P_i} \quad$ y
$\quad g_i\left(q_j, p_j\right)  =-\frac{\partial \mathcal{G}}{\partial q_i}$
	\item<7-> Si existe $\mathcal{G}\left(q_i, P_i\right)$, entonces la transformación infinitesimal es canónica.
    \end{itemize}
}

\section{Invariancia bajo transformaciones infinitesimales}
\frame{
\frametitle{Invariancia bajo transformaciones infinitesimales}
\begin{itemize}  
	\item<1->  $f_i\left(q_j, p_j\right)=\frac{\partial \mathcal{G}}{\partial P_i} = \sum_j\left(\frac{\partial \mathcal{G}}{\partial p_j} \frac{\partial p_j}{\partial P_i}+\frac{\partial \mathcal{G}}{\partial q_j} \frac{\partial q_j}{\partial P_i}\right)$ y hasta primer orden en $\epsilon$,
	$\Rightarrow f_i\left(q_j, p_j\right) =\sum_j\left(\frac{\partial \mathcal{G}}{\partial p_j} \delta_{i j}+\mathcal{O}(\epsilon)\right)  =\frac{\partial \mathcal{G}}{\partial p_i}+\mathcal{O}(\epsilon)$
	\item<2-> Consideremos el comportamiento del Hamiltoniano bajo una transformación canónica infinitesimal generada por una función $\mathcal{G}$. 
	\item<3-> El cambio en la función $K\left(q_i, p_i\right)$ definida en el espacio de fase, debido a una transformación canónica infinitesimal es
	$\delta K = K\left(Q_i, P_i\right)-K\left(q_i, p_i\right) =K\left(q_i+\epsilon f_i, p_i+\epsilon g_i\right)-K\left(q_i, p_i\right)$
	\item<4-> Desarrollando por Taylor  $\delta K  =\epsilon \sum_{i=1}^s\left(\frac{\partial K}{\partial q_i} f_i+\frac{\partial K}{\partial p_i} g_i\right)$ 
	\item<5-> Sustituyendo $f_i$ y $g_i$, 
	$\delta K =\epsilon \sum_{i=1}^s\left(\frac{\partial K}{\partial q_i} \frac{\partial \mathcal{G}}{\partial p_i}-\frac{\partial K}{\partial p_i} \frac{\partial \mathcal{G}}{\partial q_i}\right) =\epsilon\{K,  \mathcal{G}\}$
	\item<6-> Si una función $K\left(q_i, p_i\right)$ en el espacio de fase es invariante bajo una transformación canónica infinitesimal, entonces $\delta K=0$ y $\{K,  \mathcal{G}\}=0$.
\end{itemize}
}

\section{Simetrías y cantidades conservadas}
\frame{
\frametitle{Simetrías y cantidades conservadas}
\begin{itemize}  
	\item<1-> Si $K=\mathcal{H}$ entonces el cambio en el Hamiltoniano debido a una transformación canónica infinitesimal es  $\delta \mathcal{H}=\epsilon\{\mathcal{H}, \mathcal{G}\}$
	\item<2-> Si el Hamiltoniano es invariante bajo una transformación canónica infinitesimal, tendremos $\delta \mathcal{H}=0 \Leftrightarrow \{\mathcal{H}, \mathcal{G}\}=0$
	\item<3-> Por lo tanto 
	$\frac{d \mathcal{G}}{d t}=\sum_{i=1}^s\left(\frac{\partial \mathcal{G}}{\partial q_i} \dot{q}_i+\frac{\partial \mathcal{G}}{\partial p_i} \dot{p}_i\right)=\{\mathcal{G}, \mathcal{H}\} + \cancelto{0}{\frac{\partial \mathcal{G}}{\partial t}}=0 \quad \Rightarrow  \mathcal{G}\left(q_i, P_i\right)=$cte.
	\item<4-> Si el Hamiltoniano de un sistema es invariante bajo una transformación canónica infinitesimal, la función generadora de esa transformación es una cantidad conservada.
	\item<5-> Este resultado es la conexión entre simetrías y leyes de conservación para un sistema, y es equivalente al Teorema de Noether en el formalismo hamiltoniano.
	\item<6->  La relación entre invariancia y constantes de movimiento se expresa más simple en la formulación hamiltoniana.
\end{itemize}
}

\section{Ejemplo: $\mathcal{H}=q_1 p_1+q_2 p_2+a p_1^2+b p_2^2$}
\frame{
\frametitle{Ejemplo: $\mathcal{H}=q_1 p_1+q_2 p_2+a p_1^2+b p_2^2$}
\begin{itemize}  
	\item<1-> Considere un hamiltoniano $\mathcal{H}=q_1 p_1+q_2 p_2+a p_1^2+b p_2^2$
	\item<2-> Donde $f_1=p_1\left(q_1+a p_1\right)$, y  $f_2=\frac{p_1}{p_2}$ son constantes del movimiento. 
	\item<3-> Para comprobar que son constantes tenemos  
	$\left[f_1, \mathcal{H}\right] =\sum_{i=1}^s\left(\frac{\partial f_1}{\partial q_i} \frac{\partial \mathcal{H}}{\partial p_i} -\frac{\partial f_1}{\partial p_i} \frac{\partial \mathcal{H}}{\partial q_i} \right) \Rightarrow$ \\ 
	$\left[f_1, \mathcal{H}\right] = p_1\left(q_1+2 a p_1\right)-\left(q_1+2 a p_1\right) p_1=0$, del mismo modo \\
	$\left[f_2, \mathcal{H}\right] =-\frac{1}{p_2} p_1+\frac{p_1}{p_2^2} p_2=0 $. {\bf Son constantes de movimiento}
	\item<4-> Por simetría podemos proponer $f_3=p_2\left( q_2 +a p_2 \right)$ y al evaluar 
	$\left[f_3, H\right]=\sum_{i=1}^s\left(\frac{\partial f_1}{\partial q_i} \frac{\partial \mathcal{H}}{\partial p_i} -\frac{\partial f_1}{\partial p_i} \frac{\partial \mathcal{H}}{\partial q_i} \right) \Rightarrow$ \\ 
	$\left[f_3, H\right] = p_2\left(q_2+2 a p_2\right)-\left(q_2+2 a p_2\right) p_1=0$, {\bf también es constante}
	\item<5-> Para encontrar $f_4$, se pueden usar las ecuaciones de movimiento: $\dot{p}_1=-p_1 \Rightarrow p_1(t)=A e^{-t}$ con $A$ una constante de integración
	\item<6-> Por lo cual $f_4=p_1 e^t$ es una cantidad conservada
	\item<7-> También $f_5=p_2 e^t$ es una cantidad conservada pero $f_2=f_4 / f_5$
\end{itemize}
}

\begin{itemize}  
	\item<1-> El paréntesis de Poisson entre las nuevas coordenadas será
	$\{Q_i, P_j\} =\sum_{k=1}^s\left(\frac{\partial Q_i}{\partial q_k} \frac{\partial P_j}{\partial p_k}-\frac{\partial Q_i}{\partial p_k} \frac{\partial P_j}{\partial q_k}\right) =\sum_{k=1}^s \frac{\partial Q_i}{\partial q_k} \frac{\partial P_j}{\partial p_k} \Rightarrow$ \\
	$\{Q_i, P_j\} = \sum_{k=1}^s\left(\frac{\partial q_i}{\partial q_k}+\frac{\partial \dot{q}_i}{\partial q_k} d t\right)\left(\frac{\partial p_j}{\partial p_k}+\frac{\partial \dot{p}_j}{\partial p_k} d t\right) \Rightarrow$
	$\{Q_i, P_j\} =\sum_{k=1}^s\left(\delta_{i k}+\frac{\partial \dot{q}_i}{\partial q_k} d t\right)\left(\delta_{j k}+\frac{\partial \dot{p}_j}{\partial p_k} d t\right) \Rightarrow$ \\
	$\{Q_i, P_j\} =\sum_{k=1}^s\left[\delta_{i k} \delta_{j k}+\left(\delta_{j k} \frac{\partial \dot{q}_i}{\partial q_k}+\delta_{i k} \frac{\partial \dot{p}_j}{\partial p_k}\right) d t\right] \Rightarrow$ \\

	\item<2-> Por lo tanto, la transformación infinitesimal es canónica y se escribe
	$P_j =p_j\left(t_0\right)-\frac{\partial \mathcal{H}}{\partial q_i}\left(q_i\left(t_0\right), p_i\left(t_0\right)\right) d t+\cdots$ \\
	$Q_j  =q_j\left(t_0\right)+\frac{\partial \mathcal{H}}{\partial p_j}\left(q_i\left(t_0\right), p_i\left(t_0\right)\right) d t+\cdots$ 
	\item<3-> El Hamiltoniano es la función generadora de la transformación canónica infinitesimal $\left\{q_i\left(t_0\right), p_i\left(t_0\right)\right\} \rightarrow\left\{q_i\left(t_0+d t\right), p_i\left(t_0+d t\right)\right\}$
	\item<4-> La evolución temporal de un sistema en su espacio de fase es una transformación canónica inducida por el Hamiltoniano. 
\end{itemize}

\section{Transformaciones y Coordenadas Hamiltonianas Cíclicas}
\frame{
\frametitle{Coordenadas Hamiltonianas Cíclicas 1/2}
\begin{itemize}  
	\item<1-> Una transformación canónica $Q_i=Q_i\left(q_j, p_j, t\right), P_i=P_i\left(q_j, p_j, t\right)$ es conveniente cuando el cambio de variables hace cíclica alguna coordenada $Q_j$ o momento $P_k$ en el Hamiltoniano $\tilde{\mathcal{H}}\left(Q_i, P_i, t\right)$.
	\item<2-> Si $\frac{\partial \tilde{\mathcal{H}}}{\partial Q_j}=0$,  entonces, la ecuación de Hamilton para el momento $P_j$ es $\dot{P}_j=-\frac{\partial \tilde{\mathcal{H}}}{\partial Q_j}=0 \quad \Rightarrow \quad P_j=$ cte. Es decir, {\bf momento conservado}.
	\item<3-> Sea un oscilador armónico  $\mathcal{H}(q, p)=\frac{p^2}{2 m}+\frac{1}{2} k q^2=\frac{1}{2 m}\left(p^2+m^2 \omega^2 q^2\right)$
	\item<4-> Encontrar la transformación canónica $\{p, q\} \rightarrow\{P, Q\}$ con $Q$ cíclica en $\tilde{\mathcal{H}}(Q, P)$. 
	\item<5-> Sea una transformación canónica 
	$p  =f(P) \cos Q$  y $q  =\frac{f(P)}{m \omega} \operatorname{sen} Q$, entonces debemos determinar $f(P)$
	\item<6-> $\tilde{\mathcal{H}}(Q, P)=\frac{1}{2 m}\left[(p(Q, P))^2+m^2 \omega^2(q(Q, P))^2\right]=\frac{1}{2 m}[f(P)]^2$
	\item<7-> La transformación se reescribe como $p=m \omega q \cot Q \quad \Rightarrow p=p(q, Q) \Rightarrow \mathcal{F}_1(q, Q)$
\end{itemize}
}

\frametitle{Coordenadas Hamiltonianas Cíclicas 2/2}
\begin{itemize}  
	\item<1-> Para $\mathcal{F}_1$ tenemos $p=\frac{\partial \mathcal{F}_1}{\partial q}=p(q, Q) \quad$  y $\quad P=-\frac{\partial \mathcal{F}_1}{\partial Q}=P(q, Q)$
	\item<2-> Entonces $\frac{\partial \mathcal{F}_1}{\partial q}=m \omega q \cot Q$ por lo tanto $\mathcal{F}_1(q, Q)=\frac{m \omega q^2}{2} \cot Q$
	\item<3-> Por otro lado $P=-\frac{\partial \mathcal{F}_1}{\partial Q}=\frac{m \omega q^2}{2} \csc ^2 Q \Rightarrow$ $q=\sqrt{\frac{2 P}{m \omega}} \operatorname{sen} Q=q(Q, P)$
	\item<4-> Sustituyendo $q$, obtenemos $p=\sqrt{2 m \omega P} \cos Q=p(Q, P)$. 
	\item<5-> Comparando vemos que $f(P)=(2 m \omega P)^{1 / 2}$ y $\tilde{\mathcal{H}}(Q, P)=\omega P$
	\item<6-> Puesto que $\frac{\partial \mathcal{F}_1}{\partial t}=0$, tenemos $\mathcal{H}(q, p)=\tilde{\mathcal{H}}(Q, P)$
	\item<7-> Como, tenemos $\frac{\partial \mathcal{H}}{\partial t}=0  \Rightarrow \tilde{\mathcal{H}}= \mathcal{H}=E=$cte, con $E$ la energía total.
	\item<8-> Las ecuaciones de Hamilton para $Q$ y $P$ son 
	$\dot{P}=-\frac{\partial \tilde{\mathcal{H}}}{\partial Q}=0 \Rightarrow P=\mathrm{cte} \Rightarrow P=\frac{E}{\omega}$\\ $\dot{Q}=\frac{\partial \tilde{\mathcal{H}}}{\partial P}=\omega \Rightarrow Q(t)=\omega t+\varphi$
	\item<9-> Sustituyendo en $q(P, Q)$ y $p(P, Q)$ obtenemos  $q(t)=\sqrt{\frac{2 E}{m \omega^2}} \operatorname{sen} (\omega t+\varphi)\quad $ y $\quad p(t)=\sqrt{2 E m} \cos (\omega t+\varphi)$
	\item<10-> Donde $\left(\frac{2 E}{m \omega^2}\right)^{1 / 2}$ es la amplitud y $\varphi$ es la fase.

\end{itemize}

\section{Recapitulando}
\frame{
\frametitle{Recapitulando}
\begin{itemize}  
	\item<1-> {\bf Transformaciones Canónicas Infinitesimales}:
	\begin{itemize}
    	\item Forma general: $Q_i = q_i + \epsilon f_i(q_j, p_j)$, $P_i = p_i + \epsilon g_i(q_j, p_j)$
    	\item Función generadora: $F_2(q_i, P_i) = \sum q_i P_i + \epsilon G(q_i, P_i)$
    	\item Relaciones: $ f_i = \frac{\partial G}{\partial P_i}, \quad g_i = -\frac{\partial G}{\partial q_i}$
	\end{itemize}
	\item<2-> {\bf Invariancia e Invariantes}:
	\begin{itemize}
    	\item El cambio de una función $K$ del espacio de fase: $ \delta K = \epsilon \{K, G\}$
    	\item Si $K$ es invariante bajo la transformación: $\{K, G\} = 0$
	\end{itemize}
	\item<3-> {\bf Simetrías y Cantidades Conservadas}
	\begin{itemize}
    	\item Si $\mathcal{H}$ es invariante bajo la transformación: $\{\mathcal{H}, G\} = 0$
    	\item Entonces: $\frac{dG}{dt} = \{G, \mathcal{H}\} + \frac{\partial G}{\partial t} = 0 \Rightarrow G$ es  constante.
    	\item Esto es análogo al Teorema de Noether en el formalismo hamiltoniano.
	\end{itemize}
	\item<4-> {\bf Ejemplo de Hamiltoniano} $\mathcal{H} = q_1p_1 + q_2p_2 + ap_1^2 + bp_2^2$
	\begin{itemize}
    	\item Constantes del movimiento: $f_1 = p_1(q_1 + ap_1)$, $f_2 = \frac{p_1}{p_2}$
    	\item Se cumple: $\{f_1, \mathcal{H}\} = 0$, $\{f_2, \mathcal{H}\} = 0$
    	\item Por simetría: $f_3 = p_2(q_2 + bp_2)$ también es conservada.
    	\item Usando ecuaciones de movimiento: $f_4 = p_1 e^t$, $f_5 = p_2 e^t$
	\end{itemize}
\end{itemize}
}
