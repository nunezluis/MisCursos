\begin{frame}{Descripción del sistema}
\begin{itemize}
  \item Consideramos una barra rígida de masa \( M \) y longitud \( \ell \), fijada por un extremo.
  \item El centro de masa está a una distancia \( \ell/2 \) del punto de suspensión.
  \item Movimiento libre bajo gravedad en un hemisferio: péndulo esférico rígido.
\end{itemize}

\begin{frame}{Parte A: Lagrangiano y coordenadas cíclicas}
\textbf{1.} Determine el Lagrangiano del sistema. Para una barra de masa \( M \), el momento de inercia respecto al punto de suspensión es:
\[
I = \frac{1}{3} M \ell^2
\]

\textbf{2.} Escriba:
\[
L = \frac{1}{2} I (\dot{\theta}^2 + \sin^2 \theta \dot{\phi}^2) + Mg \frac{\ell}{2} \cos \theta
\]

\textbf{3.} ¿Cuál es la coordenada cíclica? ¿Qué cantidad se conserva?
\end{frame}

\begin{frame}{Parte B: Primeras integrales}
\textbf{4.} Usando la conservación de la energía, escriba la ecuación del movimiento:
\[
\frac{1}{2} I \dot{\theta}^2 + V_{\text{ef}}(\theta) = E
\]
\[
V_{\text{ef}}(\theta) = \frac{p_\phi^2}{2 I \sin^2 \theta} - Mg \frac{\ell}{2} \cos \theta
\]

\textbf{5.} Discuta el papel del momento angular \( p_\phi \) en el confinamiento del movimiento.
\end{frame}

\begin{frame}{Parte C: Movimiento en un plano}
\begin{itemize}
  \item ¿Qué condición sobre \( p_\phi \) garantiza que la barra oscile en un solo plano?
  \item Interprete físicamente \( \phi = \text{cte} \): ¿cuál es el plano del movimiento?
  \item ¿Qué rol juega la simetría azimutal rota?
\end{itemize}
\end{frame}

\begin{frame}{Objetivos del ejercicio}
\begin{itemize}
  \item Aplicar coordenadas generalizadas a un cuerpo rígido con simetría axial.
  \item Distinguir entre coordenadas cíclicas y no cíclicas.
  \item Interpretar el potencial efectivo en un sistema rotacional.
  \item Contrastar este sistema con el trompo simétrico de Lagrange.
\end{itemize}
\end{frame}
