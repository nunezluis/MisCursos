%%%%% Diapo 1
\section{Definiciones}
\frame{
  \frametitle{Definiciones}
   \begin{itemize}  
  	\item<1-> Un cuerpo rígido es un sistema de partículas cuyas distancias relativas son fijas
	\item<2-> Su movimiento se describe en términos de la posición de su centro de masa y de la orientación relativa del cuerpo en el espacio respecto al centro de masa.
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/SolidRef.png}
   	\end{figure}
	\item<3-> Esto requiere de dos sistemas de coordenadas:
   	\begin{itemize}
	\item Un sistema inercial o de laboratorio, denotado por $(x, y, z)$ y con origen en un punto fijo $O$
	\item  Un sistema en movimiento, fijo en el cuerpo, con origen en el centro de masa (CM), identificado por $\left(x_1, x_2, x_3\right)$
	\end{itemize} 
    \end{itemize}
}
%%%%% Diapo 2
\section{Desplazamiento general del cuerpo rígido}
\frame{
\frametitle{Desplazamiento general del cuerpo rígido}
\begin{itemize}  
	\item<1-> Un desplazamiento general del cuerpo rígido se representa como la suma de dos movimientos:
	\begin{itemize}  
		\item<1-> Translación del centro de masa, sin cambiar la orientación relativa entre $(x, y, z)$ y $\left(x_1, x_2, x_3\right)$.
		\item<2-> Rotación de las coordenadas $\left(x_1, x_2, x_3\right)$ alrededor de un eje que pasa por el centro de masa.
	\end{itemize}
	\item<2-> La posición $\mathbf{R}$ de un punto $P$ del cuerpo rígido con respecto al sistema de referencia del laboratorio $(x, y, z)$ es $\mathbf{R}=\mathbf{R}_{\mathrm{cm}}+\mathbf{r}$
	\item<3-> Un desplazamiento infinitesimal de $P$ será $d \mathbf{R}=d \mathbf{R}_{\mathrm{cm}}+d \mathbf{r}$
	\begin{figure}[t]
		\includegraphics[width=1.2in]{Figuras/Infinitesimal.png}
   	\end{figure}
	\item<4-> Un cambio infinitesimal $d \mathbf{r}$  sólo puede deberse a un cambio de dirección del vector $\mathbf{r}$, no a un cambio de su magnitud
\end{itemize}
}
%%%%% Diapo 3
\section{Velocidades en un cuerpo rígido}
\frame{
\frametitle{Velocidades en un cuerpo rígido}
\begin{itemize}  
	\item<1-> Si $\theta$ el ángulo entre la dirección $d \phi$ y el vector $\mathbf{r}$, entonces el vector $d \mathbf{r}$ es perpendicular al plano $(d \phi, \mathbf{r})$.
	\item<2-> Su magnitud es $d r=(r \operatorname{sen} \, \theta) d \phi$ y su dirección $d \mathbf{r}=d \mathbf{\phi} \times \mathbf{r}$
	\item<3-> La velocidad de $P$, es $\mathbf{v}_P=\mathbf{v}_{\mathrm{cm}}+\boldsymbol{\Omega} \times \mathbf{r} \Leftrightarrow \frac{d \mathbf{R}}{d t}=\frac{d \mathbf{R}_{\mathrm{cm}}}{d t}+\frac{d \phi}{d t} \times \mathbf{r}$
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/VelocidadP.png}
   	\end{figure}
	\item<4-> 
	$\mathbf{v}_P=\frac{d \mathbf{R}}{d t}$: velocidad de $P$ en el laboratorio $(x, y, z),$ \\
	$\mathbf{v}_{\mathrm{cm}}=\frac{d \mathbf{R}_{\mathrm{cm}}}{d t}$ velocidad de traslación del centro de masa en $(x, y, z)$, \\
	$\boldsymbol{\Omega}=\frac{d \boldsymbol{\phi}}{d t}$:  velocidad angular instantánea de rotación. 
	\item<5-> La dirección de la velocidad angular instantánea $\boldsymbol{\Omega}$ es la misma que la del vector $d \boldsymbol{\phi}$
\end{itemize}
}
%%%%% Diapo 4
\section{Precesión, nutación y rotación}
\frame{
\frametitle{Precesión, nutación y rotación}
\begin{itemize}  
	\item<1-> Distinguiremos tres movimientos en un cuerpo rígido
	\begin{itemize}
		\item<1-> {\bf precesión:} rotación alrededor de un eje fijo en el laboratorio, 
		\item<2-> {\bf nutación:} inclinación con respecto al eje fijo y 
		\item<3-> {\bf rotación:} rotación del cuerpo sobre sí mismo.
	\end{itemize}
	\begin{figure}[t]
		\includegraphics[width=1.0in]{Figuras/PrecesionNutacion.png}
   	\end{figure}
	\item<2-> 
	\begin{itemize}
		\item<1-> {\bf Angulo Euler de precesión}, $\phi \in[0,2 \pi]$ : ángulo de rotación con respecto al eje $z$, sobre el plano $(x, y)$,
		\item<2-> {\bf Angulo Euler de nutación}, $\theta \in[0, \pi]$ : ángulo de rotación con respecto a la línea nodal $N$, medido desde $z$ hasta $x_3$.
		\item<3-> {\bf Angulo Euler de rotación}, $\psi \in[0,2 \pi]$ : ángulo de rotación con respecto al eje $x_3$, sobre el plano $\left(x_1, x_2\right)$, medido desde $N$ a $x_1$.
	\end{itemize}
\end{itemize}
}

\section{Matrices de Rotación y el grupo ${\bf O}(n)$}
\frame{
\frametitle{Matrices de Rotación y el grupo ${\bf O}(n)$}
\begin{figure}[t]
	\includegraphics[width=1.5in]{Figuras/RotacionSimple.png}
\end{figure}
\begin{itemize}  
	\item<1-> En general
	$
	\left.
	\begin{aligned}
 		&x_1^{\prime}=x_1 \cos \phi+x_2 \operatorname{sen}  \phi \\
 		&x_2^{\prime}=-x_1 \operatorname{sen}  \phi+x_2 \cos \phi
	\end{aligned}
	\right\} \Leftrightarrow 
	\begin{pmatrix}
	x_1^{\prime} \\
	x_2^{\prime}
	\end{pmatrix} = 
	\begin{pmatrix}
	\cos \phi & \operatorname{sen}  \phi \\
	-\operatorname{sen}  \phi & \cos \phi
	\end{pmatrix}
	\begin{pmatrix}
	x_1 \\
	x_2
	\end{pmatrix}
	$
	\item<2-> Es decir, ${\bf r}^{\prime} = \mathbb{A}_\phi{\bf r}$, con $\mathbb{A}_\phi$ operadores ortogonales (Unitarios). \\ $\mathbb{A}_{-\phi} = \mathbb{A}_{\phi}^T$ y $\mathbb{A}_\phi \mathbb{A}_{\phi}^T = \mathbb{I}$. Además $\mathbb{A}_\phi \mathbb{B}_\gamma = \mathbb{C}_{\phi \gamma}$, con $\mathbb{C}_{\phi \gamma}^T = \mathbb{C}_{\phi \gamma}^{-1}$
	\item<3-> Los operadores ortogonales, $\mathbb{A}_\phi$, forman el grupo ${\bf O}(n)$
	\begin{itemize}
		\item Cerrado $\mathbb{A}_\phi$ ortogonal y  $\mathbb{B}_\gamma$ ortogonal, $\mathbb{A}_\phi \mathbb{B}_\gamma = \mathbb{C}_{\phi \gamma}$ ortogonal
		\item Inverso $\forall \mathbb{A}_\phi \quad \exists ! \; \mathbb{A}_{-\phi} \ni  \mathbb{A}_\phi \mathbb{A}_{-\phi} \equiv \mathbb{A}_\phi \mathbb{A}_{\phi}^{-1} \equiv \mathbb{A}_\phi \mathbb{A}_{\phi}^{T}= \mathbb{I}$ 
		\item Neutro $\exists ! \; \mathbb{I} \; \ni \mathbb{A}_\phi \mathbb{I} = \mathbb{I} \mathbb{A}_\phi = \mathbb{A}_\phi$
		\item Asociativa $(\mathbb{A}_\phi \mathbb{B}_\gamma) \mathbb{D}_\theta = \mathbb{A}_\phi (\mathbb{B}_\gamma \mathbb{D}_\theta)$
	\end{itemize} 
\end{itemize}
}

\section{Transformaciones entre el sistema Centro de Masa y Laboratorio}
\subsection{Rotaciones de Euler}
\frame{
\frametitle{Del sistema Centro de Masa al Laboratorio 1/2}
\begin{itemize}
	\item<1->  La transformación entre el sistema Centro de Masa y Laboratorio se construye concatenando tres transformaciones lineales independiente, cada una siguiendo un ángulo de Euler ($\phi$, $\theta$, $\psi$). 
	\item<2-> Si $r^j$ son las componentes del vector posición de un punto en respecto sistema laboratorio, $S_{xyz}$ y $\tilde{r}^i$ las  de un punto respecto al sistema centro de masa, $\tilde{S}_{x_1 x_2 x_3}$, tendremos un par de transformaciones tales que $\tilde{r}^i = \tilde{U}^i_j \, r^j $ y  $ r^j = U^j_i \, \tilde{r}^i$. Además, $\tilde{U}^i_k U^k_j = \delta^i_j \Leftrightarrow \tilde{U}^i_k = (U^i_k)^{-1}$
	\item<3-> Rotamos tres veces $\tilde{S}_{x_1,x_2,x_3}$ respecto a $S_{x,y,z}$.
	\begin{figure}[t]
		\includegraphics[width=2.4in]{Figuras/RotacionesEuler.png}
   	\end{figure}
	{\tiny $\tilde{U}^i_j(\phi)=\left(\begin{array}{ccc}\cos\phi & \operatorname{sen}\phi & 0 \\ -\operatorname{sen} \phi & \cos \phi & 0 \\ 0 & 0 & 1\end{array}\right)$, 
	$\tilde{U}^i_j(\theta)=\left(\begin{array}{ccc}1 & 0 & 0 \\ 0 & \cos \theta & \operatorname{sen} \theta \\ 0 & -\operatorname{sen} \theta & \cos \theta\end{array}\right)$  
 	$\tilde{U}^i_j(\psi)=\left(\begin{array}{ccc}\cos \psi & \operatorname{sen} \psi & 0 \\ -\operatorname{sen} \psi & \cos \psi & 0 \\ 0 & 0 & 1\end{array}\right)$ }.  

\end{itemize}
}
%%%%% Diapo 8
\subsection{Matrices de Euler}
\frame{
\frametitle{Matrices de Euler}
\begin{itemize} 
	\item<1-> En general componemos las tres rotaciones como $\tilde{\mathbb{U}}_{\psi \theta \phi} = \tilde{\mathbb{U}}_{\psi} \tilde{\mathbb{U}}_{\theta} \tilde{\mathbb{U}}_{\phi}$
	{\tiny  $\tilde{\mathbb{U}}_{\psi \theta \phi}  = \begin{pmatrix} \cos \psi \cos \phi - \operatorname{sen} \psi \cos \theta \operatorname{sen} \phi & \cos \psi \operatorname{sen} \phi + \operatorname{sen} \psi \cos \theta \cos \phi & \operatorname{sen} \psi \operatorname{sen} \theta \\ -\operatorname{sen} \psi \cos \phi - \cos \psi \cos \theta \operatorname{sen} \phi & -\operatorname{sen} \psi \operatorname{sen} \phi + \cos \psi \cos \theta \cos \phi & \cos \psi \operatorname{sen} \theta \\
	\operatorname{sen} \theta \operatorname{sen} \phi & -\operatorname{sen} \theta \cos \phi & \cos \theta \end{pmatrix}
	$}
	\item<2-> Como tenemos $\tilde{\mathbb{U}}_{\psi \theta \phi} = \tilde{\mathbb{U}}_{\psi} \tilde{\mathbb{U}}_{\theta} \tilde{\mathbb{U}}_{\phi}$ también tendremos $\tilde{\mathbb{U}}_{\psi \theta \phi}^{-1} \equiv \mathbb{U}_{\phi \theta \psi} =  \mathbb{U}_{-\phi} \mathbb{U}_{-\theta} \mathbb{U}_{-\psi}$.
	\item<3-> Por lo tanto
		{\tiny $\tilde{\mathbb{U}}_{-\phi}=\left(\begin{array}{ccc}\cos \phi & -\operatorname{sen} \phi & 0 \\ \operatorname{sen} \phi & \cos \phi & 0 \\ 0 & 0 & 1\end{array}\right)$, 
	$\tilde{\mathbb{U}}_{-\theta}=\left(\begin{array}{ccc}1 & 0 & 0 \\ 0 & \cos \theta & -\operatorname{sen} \theta \\ 0 & \operatorname{sen} \theta & \cos \theta\end{array}\right)$  
 	$\tilde{\mathbb{U}}_{-\psi}=\left(\begin{array}{ccc}\cos \psi & -\operatorname{sen} \psi & 0 \\ \operatorname{sen} \psi & \cos \psi & 0 \\ 0 & 0 & 1\end{array}\right)$ }.
	\item<4-> Finalmente tenemos que las matricies de rotación son matrices ortogonales. Esto es que la traspuesta es la inversa $\tilde{\mathbb{U}}_{\psi \theta \phi}^{-1} = \tilde{\mathbb{U}}_{\psi \theta \phi}^{T}$ y eso se ve claramente en las matrices arriba. Además la inversa de una multiplicación de matrices invierte el orden.
	\item<5-> {\tiny $\tilde{\mathbb{U}}^{-1} = \tilde{\mathbb{U}}^T = \mathbb{U} = 
	\begin{pmatrix} 
	\cos \psi \cos \phi - \operatorname{sen} \psi \cos \theta \operatorname{sen} \phi & -\operatorname{sen} \psi \cos \phi - \cos \psi \cos \theta \operatorname{sen} \phi & \operatorname{sen} \theta \operatorname{sen} \phi \\ 
	\cos \psi \operatorname{sen} \phi + \operatorname{sen} \psi \cos \theta \cos \phi & -\operatorname{sen} \psi \operatorname{sen} \phi + \cos \psi \cos \theta \cos \phi & -\operatorname{sen} \theta \cos \phi \\ \operatorname{sen} \psi \operatorname{sen} \theta & \cos \psi \operatorname{sen} \theta & \cos \theta
	\end{pmatrix}$ } 	  
\end{itemize}
}

\section{Velocidad ${\bf \Omega}$, ángulos y velocidades de Euler}
\frame{
\frametitle{Velocidad ${\bf \Omega}$ ángulos y velocidades de Euler}
\begin{itemize}
	\item<1-> Las velocidades angulares $\dot{\phi}, \dot{\theta}$ y $\dot{\psi}$ pueden expresarse en términos de sus proyecciones sobre los ejes $\left(x_1, x_2, x_3\right)$ como
	\begin{itemize}
		\item $\dot{\psi}_1  =0$, $\dot{\psi}_2  =0$ y $\dot{\psi}_3  =\dot{\psi}$;
		\item $\dot{\theta}_1  =\dot{\theta} \cos \psi$, $\dot{\theta}_2  =-\dot{\theta} \operatorname{sen} \, \psi$ y $\dot{\theta}_3  =0$,  ya que $\dot{\theta}$ es perpendicular a  $x_3$
		\item $\dot{\phi}_1=(\dot{\phi} \operatorname{sen} \, \theta) \operatorname{sen} \, \psi $,  $\dot{\phi}_2=(\dot{\phi} \operatorname{sen} \, \theta) \cos \psi$ y $\dot{\phi_3}=\dot{\phi} \cos \theta$
	\end{itemize}  
	\item<2-> La velocidad angular instantánea $\boldsymbol{\Omega}$ es una combinación de rotaciones asociadas a los tres ángulos de Euler. 
	\item<3-> Las componentes del vector $\boldsymbol{\Omega}=\left(\tilde{\Omega}_1, \tilde{\Omega}_2, \tilde{\Omega}_3\right)$ se expresan en términos de los ángulos de $(\theta, \phi, \psi)$ y de sus velocidades angulares $(\dot{\theta}, \dot{\phi}, \dot{\psi})$.
	\item<4-> Para cada componente $\tilde{\Omega}_i$, respecto al sistema centro de masa, tenemos $\tilde{\Omega}_i=\dot{\theta}_i+\dot{\phi}_i+\dot{\psi}_i, \quad i=1,2,3$. \\
	$\tilde{\Omega}_1  = \dot{\phi} \operatorname{sen} \, \theta \operatorname{sen} \, \psi+\dot{\theta} \cos \psi $ \\
	$\tilde{\Omega}_2  = \dot{\phi} \operatorname{sen} \, \theta \cos \psi-\dot{\theta} \operatorname{sen} \, \psi $ \\
	$\tilde{\Omega}_3  = \dot{\psi}+\dot{\phi} \cos \theta$
\end{itemize}
}

\subsection{Velocidades Angulares ${\bf \Omega}$ y ${\bf \tilde{\Omega}}$}
\frame{
\frametitle{Velocidades Angulares ${\bf \Omega}$ y ${\bf \tilde{\Omega}}$}
\begin{itemize}  
	\item<1-> Las matrices de Euler, $\tilde{\mathbb{U}}_{\psi \theta \phi}$, nos permiten calcular las componentes de la velocidad angular respecto al sistema laboratorio $\Omega^1, \Omega^2, \Omega^3$ a partir de las componentes respecto al sistema CM $\tilde{\Omega}^1, \tilde{\Omega}^2, \tilde{\Omega}^3$.
	\item<2-> {\tiny
	$
	\begin{pmatrix} 
		\Omega^1 \\ \Omega^2 \\ \Omega^3 
	\end{pmatrix} = 
	\begin{pmatrix} 
	\cos \psi \cos \phi - \operatorname{sen} \psi \cos \theta \operatorname{sen} \phi & -\operatorname{sen} \psi \cos \phi - \cos \psi \cos \theta \operatorname{sen} \phi & \operatorname{sen} \theta \operatorname{sen} \phi \\ 
	\cos \psi \operatorname{sen} \phi + \operatorname{sen} \psi \cos \theta \cos \phi & -\operatorname{sen} \psi \operatorname{sen} \phi + \cos \psi \cos \theta \cos \phi & -\operatorname{sen} \theta \cos \phi \\ 
	\operatorname{sen} \psi \operatorname{sen} \theta & \cos \psi \operatorname{sen} \theta & \cos \theta 
	\end{pmatrix} 
	\begin{pmatrix} 
		\tilde{\Omega}^1 \\ \tilde{\Omega}^2 \\ \tilde{\Omega}^3 
	\end{pmatrix}
	$ 
	}
	\item<3-> Dado que para el centro de masa \\
	$\tilde{\Omega}^1  = \dot{\phi} \operatorname{sen} \, \theta \operatorname{sen} \, \psi+\dot{\theta} \cos \psi $ \\
	$\tilde{\Omega}^2  = \dot{\phi} \operatorname{sen} \, \theta \cos \psi-\dot{\theta} \operatorname{sen} \, \psi $ \\
	$\tilde{\Omega}^3  = \dot{\psi}+\dot{\phi} \cos \theta$
	\item<4-> Tendremos para el sistema laboratorio \\
	$\Omega^1 = \dot{\theta} \cos \phi+\dot{\psi} \operatorname{sen} \phi \operatorname{sen} \theta$ \\
	$\Omega^2 =\dot{\theta} \operatorname{sen} \phi-\dot{\psi} \cos \phi \operatorname{sen} \theta$ \\
	$\Omega^3 =\dot{\phi}+\dot{\psi} \cos \theta$	
\end{itemize}
}

	${\bf \hat{z}} \cdot {\bf \hat{x}_1} = \operatorname{sen} \psi \operatorname{sen} \theta$;
	$\quad {\bf \hat{z}} \cdot {\bf \hat{x}_2} = \cos \psi \operatorname{sen} \theta$;
	$\quad {\bf \hat{z}} \cdot {\bf \hat{x}_3} = \cos \theta$. 
\end{itemize}

\subsection{Vectores unitarios}
\frame{
\frametitle{Vectores unitarios}
\begin{itemize}
	\item<1-> Más aún si ${\bf \hat{x}} _1 = \alpha {\bf \hat{x}}  + \beta {\bf \hat{y}}  + \gamma {\bf \hat{z}} $ y al proyectarla tendremos que ${\bf \hat{x}} _1 = ({\bf \hat{x}} \cdot {\bf \hat{x}_1}) {\bf \hat{x}}  + ({\bf \hat{y}} \cdot {\bf \hat{x}_1}) {\bf \hat{y}}  + ({\bf \hat{z}} \cdot {\bf \hat{x}_1}) {\bf \hat{z}} $. Por lo tanto expresamos 
	${\bf \hat{x}} _1 = (\cos \psi \cos \phi - \operatorname{sen} \psi \cos \theta \operatorname{sen} \phi) {\bf \hat{x}}  + (\cos \psi \operatorname{sen} \phi + \operatorname{sen} \psi \cos \theta \cos \phi) {\bf \hat{y}}  + (\operatorname{sen} \psi \operatorname{sen} \theta) {\bf \hat{z}} $
	\item<2-> Del mismo modo ${\bf \hat{x}} _2 = ({\bf \hat{x}} \cdot {\bf \hat{x}_2}) {\bf \hat{x}}  + ({\bf \hat{y}} \cdot {\bf \hat{x}_2}) {\bf \hat{y}}  + ({\bf \hat{z}} \cdot {\bf \hat{x}_2}) {\bf \hat{z}} $. Con lo cual 
	${\bf \hat{x}} _2 = (-\operatorname{sen} \psi \cos \phi - \cos \psi \cos \theta \operatorname{sen} \phi){\bf \hat{x}} + (-\operatorname{sen} \psi \operatorname{sen} \phi + \cos \psi \cos \theta \cos \phi){\bf \hat{y}} + \cos \psi \operatorname{sen} \theta{\bf \hat{z}}$ \\
	\item<3-> Finalmente 
	${\bf \hat{x}} _3 = ({\bf \hat{x}} \cdot {\bf \hat{x}_3}) {\bf \hat{x}}  + ({\bf \hat{y}} \cdot {\bf \hat{x}_3}) {\bf \hat{y}}  + ({\bf \hat{z}} \cdot {\bf \hat{x}_3}) {\bf \hat{z}}$. Finalmente, ${\bf \hat{x}} _3 = \operatorname{sen} \theta \operatorname{sen} \phi {\bf \hat{x}}  -\operatorname{sen} \theta \cos \phi {\bf \hat{y}}  + \cos \theta {\bf \hat{z}}$
 
\end{itemize}

\section{Recapitulando}
\frame{
\frametitle{Recapitulando}
\begin{itemize}  
	\item<1-> \textbf{Cuerpo rígido}: sistema de partículas con distancias relativas fijas.
	\item<2-> El movimiento se descompone en: \textbf{Traslación} centro de masa; \textbf{Rotación} alrededor del centro de masa
	\item<3-> \textbf{Sistemas Laboratorio/Inercial ($x, y, z$)}; \\ \textbf{Sistema del centro de masa ($x_1, x_2, x_3$)}.
	\item<4-> \textbf{Descripción de la orientación respecto al CM:} Se utilizan los \textbf{ángulos de Euler} \( (\phi, \theta, \psi) \):
    	\begin{itemize}
        		\item \(\phi\) (precesión): rotación alrededor del eje z del Sistema Laboratorio
        		\item \(\theta\) (nutación): inclinación del eje z al eje \(x_3\)
        		\item \(\psi\) (rotación): giro alrededor del eje \(x_3\)
    	\end{itemize}
	\item<5-> {Velocidad angular \({\bf \Omega}\):} Velocidad de un punto: ${\bf v}_P = {\bf v}_{CM} + {\bf \Omega} \times {\bf r}$
	\item<6-> \textbf{\({\bf \Omega}\)} en el sistema CM:
	$\begin{cases}
	\tilde{\Omega}_1 = \dot{\phi} \operatorname{sen}  \theta \operatorname{sen}  \psi + \dot{\theta} \cos \psi \\
	\tilde{\Omega}_2 = \dot{\phi} \operatorname{sen}  \theta \cos \psi - \dot{\theta} \operatorname{sen}  \psi \\
	\tilde{\Omega}_3 = \dot{\psi} + \dot{\phi} \cos \theta
	\end{cases}$
\end{itemize}

\section*{Planteamiento del ejercicio}

Un rotor simétrico (por ejemplo, un trompo ideal) tiene momentos de inercia respecto a sus ejes principales dados por:
\[
I_1 = I_2 = I \neq I_3.
\]
El rotor gira libremente en el espacio, sin estar sometido a torques externos. Su orientación se describe mediante los ángulos de Euler \( (\phi, \theta, \psi) \). Inicialmente, el cuerpo gira con una velocidad angular constante \( \dot{\psi}_0 \) alrededor del eje de simetría \( x_3 \), formando un ángulo fijo \( \theta_0 \) con respecto al eje vertical \( z \).

\vspace{0.5em}
\noindent
\textbf{Objetivo:} Analizar el movimiento rotacional del rotor y determinar las cantidades conservadas y la evolución temporal de los ángulos de Euler.

\vspace{1em}
\noindent
\textbf{Preguntas:}
\begin{enumerate}
    \item \textbf{(Cinemática)} Escribe la expresión de la velocidad angular \( \vec{\Omega} \) del rotor en el sistema del cuerpo, en términos de los ángulos de Euler.

    \item \textbf{(Dinámica)} Escribe las ecuaciones de Euler para un cuerpo rígido simétrico y deduce sus consecuencias para este caso de rotación libre.

    \item \textbf{(Conservación)} Demuestra que:
    \begin{itemize}
        \item La componente \( L_z \) del momento angular en el sistema del laboratorio es constante.
        \item La energía cinética total también es constante.
    \end{itemize}

    \item \textbf{(Solución del movimiento)} Suponiendo que el ángulo \( \theta = \theta_0 \) es constante:
    \begin{itemize}
        \item Demuestra que \( \dot{\phi} \) es constante y \( \dot{\theta} = 0 \).
        \item Obtén la relación entre \( \dot{\phi} \), \( \dot{\psi}_0 \), \( I \), \( I_3 \) y \( \theta_0 \).
    \end{itemize}
\end{enumerate}

\vspace{1em}
\noindent
\textbf{Sugerencia:} Usa la representación de la velocidad angular en términos de ángulos de Euler en la convención \( z \to x' \to z'' \) y trabaja en el sistema del cuerpo.

\vspace{1em}
\noindent
\textbf{Datos útiles:}
\begin{itemize}
    \item El sistema está libre de torques externos.
    \item El momento de inercia es constante en el sistema del cuerpo.
    \item El sistema de referencia inercial está orientado con el eje vertical \( z \).
\end{itemize}
