%%%%% Diapo 1
\section{Definiciones}
\frame{
  \frametitle{Definiciones}
   \begin{itemize}  
  	\item<1-> Un cuerpo rígido es un sistema de partículas cuyas distancias relativas son fijas
	\item<2-> Su movimiento se describe en términos de la posición de su centro de masa y de la orientación relativa del cuerpo en el espacio
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/SolidRef.png}
   	\end{figure}
	\item<3-> Esto requiere de dos sistemas de coordenadas:
   	\begin{itemize}
	\item Un sistema inercial o laboratorio, denotado por $(x, y, z)$ y con origen en un punto fijo $O$
	\item  Un sistema en movimiento, fijo en el cuerpo, con origen en el centro de masa (CM), identificado por $\left(x_1, x_2, x_3\right)$
	\end{itemize} 
    \end{itemize}
}
%%%%% Diapo 2
\section{Desplazamiento general del cuerpo rígido}
\frame{
\frametitle{Desplazamiento general del cuerpo rígido}
\begin{itemize}  
	\item<1-> Un desplazamiento general del cuerpo rígido se representa como la suma de dos movimientos:
	\begin{itemize}  
		\item<1-> Translación del centro de masa, sin cambiar la orientación relativa entre $(x, y, z)$ y $\left(x_1, x_2, x_3\right)$.
		\item<2-> Rotación de las coordenadas $\left(x_1, x_2, x_3\right)$ alrededor de un eje que pasa por el centro de masa.
	\end{itemize}
	\item<2-> La posición $\mathbf{R}$ de un punto $P$ del cuerpo rígido con respecto al sistema de referencia del laboratorio $(x, y, z)$ es $\mathbf{R}=\mathbf{R}_{\mathrm{cm}}+\mathbf{r}$
	\item<3-> Un desplazamiento infinitesimal de $P$ será $d \mathbf{R}=d \mathbf{R}_{\mathrm{cm}}+d \mathbf{r}$
	\begin{figure}[t]
		\includegraphics[width=1.2in]{Figuras/Infinitesimal.png}
   	\end{figure}
	\item<4-> Un cambio infinitesimal $d \mathbf{r}$  sólo puede deberse a un cambio de dirección del vector $\mathbf{r}$, no a un cambio de su magnitud
\end{itemize}
}
%%%%% Diapo 2
\section{Velocidades en un cuerpo rígido}
\frame{
\frametitle{Velocidades en un cuerpo rígido}
\begin{itemize}  
	\item<1-> Si $\theta$ el ángulo entre la dirección $d \phi$ y el vector $\mathbf{r}$, entonces el vector $d \mathbf{r}$ es perpendicular al plano $(d \phi, \mathbf{r})$.
	\item<2-> Su magnitud es $d r=(r \operatorname{sen} \, \theta) d \phi$ y su dirección $d \mathbf{r}=d \mathbf{\phi} \times \mathbf{r}$
	\item<3-> La velocidad de $P$, es $\mathbf{v}_P=\mathbf{v}_{\mathrm{cm}}+\boldsymbol{\Omega} \times \mathbf{r} \Leftrightarrow \frac{d \mathbf{R}}{d t}=\frac{d \mathbf{R}_{\mathrm{cm}}}{d t}+\frac{d \phi}{d t} \times \mathbf{r}$
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/VelocidadP.png}
   	\end{figure}
	\item<4-> 
	$\mathbf{v}_P=\frac{d \mathbf{R}}{d t}$: velocidad de $P$ en el laboratorio $(x, y, z),$ \\
	$\mathbf{v}_{\mathrm{cm}}=\frac{d \mathbf{R}_{\mathrm{cm}}}{d t}$ velocidad de traslación del centro de masa en $(x, y, z)$, \\
	$\boldsymbol{\Omega}=\frac{d \boldsymbol{\phi}}{d t}$:  velocidad angular instantánea de rotación. 
	\item<5-> La dirección de la velocidad angular instantánea $\boldsymbol{\Omega}$ es la misma que la del vector $d \boldsymbol{\phi}$
\end{itemize}
}
%%%%% Diapo 2
\section{Precesión, nutación y rotación}
\frame{
\frametitle{Precesión, nutación y rotación}
\begin{itemize}  
	\item<1-> Distinguiremos tres movimientos en un cuerpo rígido
	\begin{itemize}
		\item<1-> {\bf precesión:} rotación alrededor de un eje fijo en el laboratorio, 
		\item<2-> {\bf nutación:} inclinación con respecto al eje fijo y 
		\item<3-> {\bf rotación:} rotación del cuerpo sobre sí mismo.
	\end{itemize}
	\begin{figure}[t]
		\includegraphics[width=1.0in]{Figuras/PrecesionNutacion.png}
   	\end{figure}
	\item<2-> 
	\begin{itemize}
		\item<1-> {\bf Angulo Euler de precesión}, $\phi \in[0,2 \pi]$ : ángulo de rotación con respecto al eje $z$, sobre el plano $(x, y)$,
		\item<2-> {\bf Angulo Euler de nutación}, $\theta \in[0, \pi]$ : ángulo de rotación con respecto a la línea nodal $N$, medido desde $z$ hasta $x_3$.
		\item<3-> {\bf Angulo Euler de rotación}, $\psi \in[0,2 \pi]$ : ángulo de rotación con respecto al eje $x_3$, sobre el plano $\left(x_1, x_2\right)$, medido desde $N$ a $x_1$.
	\end{itemize}
\end{itemize}
}
%%%%% Diapo 2
\section{Velocidad ${\bf \Omega}$, ángulos y velocidades de Euler}
\frame{
\frametitle{Velocidad ${\bf \Omega}$ ángulos y velocidades de Euler}
\begin{itemize}
	\item<1-> Las velocidades angulares $\dot{\phi}, \dot{\theta}$ y $\dot{\psi}$ pueden expresarse en términos de sus proyecciones sobre los ejes $\left(x_1, x_2, x_3\right)$ como
	\begin{itemize}
		\item $\dot{\psi}_1  =0$, $\dot{\psi}_2  =0$ y $\dot{\psi}_3  =\dot{\psi}$;
		\item $\dot{\theta}_1  =\dot{\theta} \cos \psi$, $\dot{\theta}_2  =-\dot{\theta} \operatorname{sen} \, \psi$ y $\dot{\theta}_3  =0$,  ya que $\dot{\theta}$ es perpendicular a  $x_3$
		\item $\dot{\phi}_1=(\dot{\phi} \operatorname{sen} \, \theta) \operatorname{sen} \, \psi $,  $\dot{\phi}_2=(\dot{\phi} \operatorname{sen} \, \theta) \cos \psi$ y $\dot{\phi_3}=\dot{\phi} \cos \theta$
	\end{itemize}  
	\item<2-> La velocidad angular instantánea $\boldsymbol{\Omega}$ es una combinación de rotaciones asociadas a los tres ángulos de Euler. 
	\item<3-> Las componentes del vector $\boldsymbol{\Omega}=\left(\Omega_1, \Omega_2, \Omega_3\right)$ se expresan en términos de los ángulos de $(\theta, \phi, \psi)$ y de sus velocidades angulares $(\dot{\theta}, \dot{\phi}, \dot{\psi})$.
	\item<4-> Para cada componente $\Omega_i$, respecto al sistema centro de masa, tenemos $\Omega_i=\dot{\theta}_i+\dot{\phi}_i+\dot{\psi}_i, \quad i=1,2,3$. \\
	$\Omega_1  = \dot{\phi} \operatorname{sen} \, \theta \operatorname{sen} \, \psi+\dot{\theta} \cos \psi $ \\
	$\Omega_2  = \dot{\phi} \operatorname{sen} \, \theta \cos \psi-\dot{\theta} \operatorname{sen} \, \psi $ \\
	$\Omega_3  = \dot{\psi}+\dot{\phi} \cos \theta$
\end{itemize}
}

\section{Transformaciones entre el sistema Centro de Masa y Laboratorio}
\frame{
\frametitle{Del sistema Centro de Masa al Laboratorio 1/2}
\begin{itemize}
	\item<1->  La transformación entre el sistema Centro de Masa y Laboratorio se construye concatenando tres transformaciones lineales independiente, cada una siguiendo un ángulo de Euler ($\phi$, $\theta$, $\psi$). 
	\item<2-> Si $\mathbf{r}_L$ son las coordenadas de un punto en el sistema $S_{xyz}$ laboratorio y $\tilde{\mathbf{r}}_{cm}$ las  de un punto en el sistema $\tilde{S}_{x_1 x_2 x_3}$ centro de masa, queremos un par de transformaciones tales que 
	$\tilde{\mathbf{r}}_{cm} = \tilde{\mathbb{U}}\, \mathbf{r}_L \Leftrightarrow \tilde{r}^i = \tilde{U}^i_j \, r^j \quad$ y 
	$\quad \mathbf{r}_{L} = \mathbb{U}\, \mathbf{r}_{cm} \Leftrightarrow r^i = U^i_j \, \tilde{r}^j$ 
	\item<3> Claramente $\mathbb{U}\tilde{\mathbb{U}} \equiv \tilde{\mathbb{U}}\mathbb{U} = \mathbb{I} \Rightarrow \tilde{\mathbb{U}} = \mathbb{U}^{-1}$ 
	\item<4-> Rotamos tres veces $\tilde{S}_{x_1,x_2,x_3}$ respecto a $S_{x,y,z}$.
	\begin{figure}[t]
		\includegraphics[width=2.5in]{Figuras/RotacionesEuler.png}
   	\end{figure}
	{\tiny $\tilde{\mathbb{U}}_\phi=\left(\begin{array}{ccc}\cos \phi & \operatorname{sen} \phi & 0 \\ -\operatorname{sen} \phi & \cos \phi & 0 \\ 0 & 0 & 1\end{array}\right)$, 
	$\tilde{\mathbb{U}}_\theta=\left(\begin{array}{ccc}1 & 0 & 0 \\ 0 & \cos \theta & \operatorname{sen} \theta \\ 0 & -\operatorname{sen} \theta & \cos \theta\end{array}\right)$  
 	$\tilde{\mathbb{U}}_\psi=\left(\begin{array}{ccc}\cos \psi & \operatorname{sen} \psi & 0 \\ -\operatorname{sen} \psi & \cos \psi & 0 \\ 0 & 0 & 1\end{array}\right)$ }.  

\end{itemize}
}
%%%%% Diapo 2
\section{Sección}
\frame{
\frametitle{Título transparencia}
\begin{itemize} 
	\item<1-> En general concatenamos las tres rotaciones como $\tilde{\mathbb{U}}_{\psi \theta \phi} = \tilde{\mathbb{U}}_{\psi} \tilde{\mathbb{U}}_{\theta} \tilde{\mathbb{U}}_{\phi}$
	{\tiny  $\tilde{\mathbb{U}}_{\psi \theta \phi}  = \begin{pmatrix} \cos \psi \cos \phi - \operatorname{sen} \psi \cos \theta \operatorname{sen} \phi & \cos \psi \operatorname{sen} \phi + \operatorname{sen} \psi \cos \theta \cos \phi & \operatorname{sen} \psi \operatorname{sen} \theta \\ -\operatorname{sen} \psi \cos \phi - \cos \psi \cos \theta \operatorname{sen} \phi & -\operatorname{sen} \psi \operatorname{sen} \phi + \cos \psi \cos \theta \cos \phi & \cos \psi \operatorname{sen} \theta \\
	\operatorname{sen} \theta \operatorname{sen} \phi & -\operatorname{sen} \theta \cos \phi & \cos \theta \end{pmatrix}
	$}
	\item<2-> Como tenemos $\tilde{\mathbb{U}}_{\psi \theta \phi} = \tilde{\mathbb{U}}_{\psi} \tilde{\mathbb{U}}_{\theta} \tilde{\mathbb{U}}_{\phi}$ también tendremos $\tilde{\mathbb{U}}_{\psi \theta \phi}^{-1} \equiv \mathbb{U}_{\phi \theta \psi} =  \mathbb{U}_{-\phi} \mathbb{U}_{-\theta} \mathbb{U}_{-\psi}$.
	\item<3-> Por lo tanto
		{\tiny $\tilde{\mathbb{U}}_{-\phi}=\left(\begin{array}{ccc}\cos \phi & -\operatorname{sen} \phi & 0 \\ \operatorname{sen} \phi & \cos \phi & 0 \\ 0 & 0 & 1\end{array}\right)$, 
	$\tilde{\mathbb{U}}_{-\theta}=\left(\begin{array}{ccc}1 & 0 & 0 \\ 0 & \cos \theta & -\operatorname{sen} \theta \\ 0 & \operatorname{sen} \theta & \cos \theta\end{array}\right)$  
 	$\tilde{\mathbb{U}}_{-\psi}=\left(\begin{array}{ccc}\cos \psi & -\operatorname{sen} \psi & 0 \\ \operatorname{sen} \psi & \cos \psi & 0 \\ 0 & 0 & 1\end{array}\right)$ }.
	\item<4-> Finalmente tenemos que las matricies de rotación son matrices ortogonales. Esto es que la traspuesta es la inversa $\tilde{\mathbb{U}}_{\psi \theta \phi}^{-1} = \tilde{\mathbb{U}}_{\psi \theta \phi}^{T}$ y eso se ve claramente en las matrices arriba. Además la inversa de una multiplicación de matrices invierte el orden.
	\item<5-> {\tiny $\tilde{\mathbb{U}}^{-1} = \tilde{\mathbb{U}}^T = 
	\begin{pmatrix} 
	\cos \psi \cos \phi - \operatorname{sen} \psi \cos \theta \operatorname{sen} \phi & -\operatorname{sen} \psi \cos \phi - \cos \psi \cos \theta \operatorname{sen} \phi & \operatorname{sen} \theta \operatorname{sen} \phi \\ 
	\cos \psi \operatorname{sen} \phi + \operatorname{sen} \psi \cos \theta \cos \phi & -\operatorname{sen} \psi \operatorname{sen} \phi + \cos \psi \cos \theta \cos \phi & -\operatorname{sen} \theta \cos \phi \\ \operatorname{sen} \psi \operatorname{sen} \theta & \cos \psi \operatorname{sen} \theta & \cos \theta
	\end{pmatrix}$ } 

	  
\end{itemize}
}
