\section{Moneda que rueda sin deslizar}
\subsection{Ligaduras}
\frame{
  \frametitle{Moneda que rueda sin deslizar: Ligaduras}
Un disco homogéneo (una moneda) de radio $a$ y masa $M$ rueda sin deslizar por una superficie plana. Escriba las ecuaciones de movimiento y encuentre una solución en el caso en que la inclinación del disco sea constante.
Algunas animaciones ilustrativas en \url{https://rotations.berkeley.edu/the-rolling-disk/\#Korteweg}
\begin{figure}[t]
	\includegraphics[width=2.8in]{Figuras/Moneda.pdf}
\end{figure}
   \begin{itemize}  
  	\item<1-> En principio tendremos seis grados de libertad $(x, y, z, \phi, \psi, \theta)$: \\ tres de traslación y tres ángulos de Euler. 
	\item<3->  Rodar sin deslizar implica que la velocidad del punto $p$, en contacto con la superficie, es instantáneamente cero.
	\item<4-> Esto es: ${\bf r}_{op}={\bf R}+{\bf r}_{cp} \Rightarrow \dot{{\bf r}}_{op} = 0=\dot{{\bf R}}+  \tilde{ {\bf \Omega}} \times {\bf r}_{cp}$. 
    \end{itemize}
}
%%%%% Diapo 2
\subsection{El Lagrangeano}
\frame{
\frametitle{Moneda que rueda sin deslizar: Lagrangeano}
\begin{itemize}
	\item<1-> Como ${\bf r}_{cp} = -a {\bf \hat{x}}_2$, al proyectar la primera ecuación tendremos $x_{op} = x   -a {\bf \hat{x}} \cdot {\bf \hat{x}}_2$, $y_{op} = y   -a {\bf \hat{y}} \cdot {\bf \hat{x}}_2$ y $0 = z   -a {\bf \hat{z}} \cdot {\bf \hat{x}}_2$.
	\item<2->  Que se convierten en  $x_{op} = x   +a (\operatorname{sen} \psi \cos \phi + \cos \psi \cos \theta \operatorname{sen} \phi)$;  $y_{op} = y   +a (\operatorname{sen} \psi \operatorname{sen} \phi - \cos \psi \cos \theta \cos \phi)$ y $z_{op} = 0 = z   -a \cos \psi \operatorname{sen} \theta$
	\item<3-> Por su parte, $\tilde{{\bf \Omega}} \times {\bf r}_{cp} = 
	\left|
	\begin{array}{ccc} 
	{\bf \hat{x}}_1 & {\bf \hat{x}}_2 & {\bf \hat{x}}_3 \\
	\tilde{\Omega}^1 & \tilde{\Omega}^2 & {\tilde \Omega}^3 \\
	0 & -a & 0
	\end{array}\right| = a(\tilde{\Omega}^3 \, {\bf \hat{x}}_1 -\tilde{\Omega}^1\, {\bf \hat{x}}_3) $
	\item<4-> Al proyectar la ecuación de ligadura $0=\dot{{\bf R}}+ a(\tilde{\Omega}^3 \, {\bf \hat{x}}_1 -\tilde{\Omega}^1\, {\bf \hat{x}}_3)$ respecto a la base del Sistema laboratorio tendremos \\
	$\dot{x} + a(\tilde{\Omega}^3 ({\bf \hat{x}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{x}} \cdot {\bf \hat{x}}_3))=0 $; $\quad \dot{y} + a(\tilde{\Omega}^3  ({\bf \hat{y}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{y}} \cdot {\bf \hat{x}}_3)) =0$;  $\quad \dot{z} + a(\tilde{\Omega}^3  ({\bf \hat{z}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{z}} \cdot {\bf \hat{x}}_3)) =0 $.
	\item<5-> Es decir $ \dot{x}+a \dot{\phi} \cos \phi \cos \theta-a \dot{\theta} \operatorname{sen} \phi \operatorname{sen} \theta+a \dot{\psi} \cos \phi  =0$; $ \dot{y}+a \dot{\phi} \operatorname{sen} \phi \cos \theta+a \dot{\theta} \cos \phi \operatorname{sen} \theta+a \dot{\psi} \operatorname{sen} \phi  =0 $;  $\quad \dot{z}-a \dot{\theta} \cos \theta =0$. Nótese que hemos tomado $\psi = 0$ por la simetría respecto al eje ${\bf \hat{x}}_3$.  
\end{itemize} 
}
%%%%% Diapo 3
\subsection{Energías cinética y potencial}
\frame{
\frametitle{Energías cinética y potencial}
\begin{itemize}
	\item<1-> La tercera ecuación es integrable y nos da $z=a \operatorname{sen} \theta$,  las otras ecuaciones constituyen vínculos no integrables.
	\item<2-> La energía cinética será 	$T= T_{tras} + T_{rot} = \frac{1}{2} M(\dot{x}^2+\dot{y}^2+\dot{z}^2)
	+\frac{1}{2}( I_{11} \tilde{\Omega}_1^2 +I_{22} \tilde{\Omega}_2^2 +I_{33} \tilde{\Omega}_3^2 )$ 
	\item<3-> Como  $\tilde{\Omega}^1  = \dot{\theta} $ $\quad \tilde{\Omega}^2  = \dot{\phi} \operatorname{sen} \, \theta \quad $  y  $\quad \tilde{\Omega}^3  = \dot{\psi}+\dot{\phi} \cos \theta$
	\item<4-> Adicionalmente, $I_{11} = I_{22}=\frac{1}{4} M a^2$ y $I_{33}=\frac{1}{2} M a^2$.	
	\item<5->  Tendremos $T=\frac{1}{2} M\left(\dot{x}^2+\dot{y}^2+\dot{z}^2\right)$ $+\frac{1}{8} M a^2\left[\dot{\phi}^2 \operatorname{sen}^2 \theta+\dot{\theta}^2+2\left(\dot{\phi} \cos \theta^2+\dot{\psi}\right)^2\right]$
	\item<6-> Que se simplifica a cuando incorporamos las ligaduras  $T = \frac{1}{32} a^2 M \left(\dot{\phi}^2 (\cos 4 \theta+13) +32 \dot{\phi} \dot{\psi} \cos \theta +2 \dot{\phi} (5 \dot{\phi} +4 \dot{\psi}) \cos 2 \theta \right.$ $\qquad \left.+8 \dot{\phi} \dot{\psi}+24 \dot{\psi}^2+20 \dot{\theta}^2\right)$
	\item<7-> Por su parte, la energía potencial $V =mg z = mga \operatorname{sen} \theta$
\end{itemize}
}

\section{Moneda en un plano inclinado}
\subsection{Planteamiento del problema}
\frame{
\frametitle{Planteamiento del problema}
Un disco delgado, uniforme, de masa $M$ y radio a está enganchado a una varilla $A C$ sin masa de longitud $b$. El sistema está en un plano inclinado perfectamente rugoso que forma un ángulo $\alpha$ con la horizontal. El punto $A$ de la varilla se mantiene fijo en un punto $O$ del plano inclinado mientras que el disco puede rodar libremente sin deslizar. Tomamos como sistema $S$ uno con origen en $O$, eje $z$ perpendicular al plano inclinado y el eje $y$ hacia arriba del plano y para el sistema $\tilde{S}$ origen también en $O$ y eje $\tilde{3}$ en la direción $A C$. %%%%% Diapo 6
\subsection{Coordenadas y ligaduras}
\frame{
\frametitle{Coordenadas y ligaduras}
\begin{itemize}  
	\item<1->  Tenemos seis grados de libertad $(x, y, z, \phi, \psi, \theta)$ y cinco ligaduras
	\item<2-> $\theta=$ constante implica  $\theta+\beta=\frac{\pi}{2}, \quad \tan \beta=\frac{a}{b}, \quad \theta=\frac{\pi}{2} -\arctan \frac{a}{b}$
	\item<3-> $|{\bf R}| = b$. Más aún, ${\bf R} =b {\bf \hat{x}}_3= b\left[\operatorname{sen} \phi \operatorname{sen} \theta {\bf \hat{x}}-\cos \phi \operatorname{sen} \theta {\bf \hat{y}}+\cos \theta {\bf \hat{z}}\right]$,
	\item<4-> Entonces $x = b\operatorname{sen} \phi \operatorname{sen} \theta $; $\quad y =-b\cos \phi \operatorname{sen} \theta \quad$ y $z = b\cos \theta $ que cumplen con la ecuación $x^2 +y^2 +z^2 = b^2$
	\item<5-> Derivándolas con $\theta=$ const., se obtiene: $
\dot{x}=b \dot{\phi} \cos \phi \operatorname{sen} \theta, \quad \dot{y}=b \dot{\phi} \operatorname{sen} \phi \operatorname{sen} \theta, \quad \dot{z}=0$
	\item<6-> Rodar sin deslizar implica que la velocidad del punto $p$ es cero : ${\bf r}_{op}={\bf R}+{\bf r}_{cp} \Rightarrow \dot{{\bf r}}_{op} = 0=\dot{{\bf R}}+  \tilde{ {\bf \Omega}} \times {\bf r}_{cp}$.
	\item<7-> Respecto al sistema centro de masa, $\tilde{S}$ tenemos ${\bf r}_{cp} = -a {\bf \hat{x}}_2$, y $\tilde{{\bf \Omega}} \times {\bf r}_{cp} = 
	\left|
	\begin{array}{ccc} 
	{\bf \hat{x}}_1 & {\bf \hat{x}}_2 & {\bf \hat{x}}_3 \\
	\tilde{\Omega}^1 & \tilde{\Omega}^2 & {\tilde \Omega}^3 \\
	0 & -a & 0
	\end{array}\right| = a(\tilde{\Omega}^3 \, {\bf \hat{x}}_1 -\tilde{\Omega}^1\, {\bf \hat{x}}_3) $
	\item<8-> Con cual  $b \dot{\hat{{\bf x}}}_3 +a(\tilde{\Omega}^3 \, {\bf \hat{x}}_1 -\tilde{\Omega}^1\, {\bf \hat{x}}_3) =0$

\end{itemize}
}
%%%%% Diapo 7
\subsection{Integración de ligaduras}
\frame{
\frametitle{Integración de ligaduras}
\begin{itemize}
	\item<1-> Al proyectar la ecuación de ligadura $0=\dot{{\bf R}}+ a(\tilde{\Omega}^3 \, {\bf \hat{x}}_1 -\tilde{\Omega}^1\, {\bf \hat{x}}_3) $ 
	respecto a la base del Sistema laboratorio tendremos \\
	$\dot{x} + a(\tilde{\Omega}^3 ({\bf \hat{x}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{x}} \cdot {\bf \hat{x}}_3))=0 $; $\quad \dot{y} + a(\tilde{\Omega}^3  ({\bf \hat{y}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{y}} \cdot {\bf \hat{x}}_3)) =0$; $\quad \dot{z} + a(\tilde{\Omega}^3  ({\bf \hat{z}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{z}} \cdot {\bf \hat{x}}_3)) =0 $.
	\item<2-> que se convierte en 
	$ \dot{x}+a(\dot{\psi} \cos \phi+\dot{\phi} \cos \phi \cos \theta-\dot{\theta} \operatorname{sen} \phi \operatorname{sen} \theta)=0$,  \\
	$ \dot{y}+a(\dot{\psi} \operatorname{sen} \phi+\dot{\phi} \operatorname{sen} \phi \cos \theta+\dot{\theta} \cos \phi \operatorname{sen} \theta)=0 $ y  
	$\quad  \dot{z}-a \dot{\theta} \cos \theta=0$. Hemos tomado $\psi = 0$ por la simetría del cuerpo respecto a ${\bf \hat{x}}_3$. 
	\item<3-> y usando las otras ecuaciones de ligadura $\dot{x}=b \dot{\phi} \cos \phi \operatorname{sen} \theta, \quad \dot{y}=b \dot{\phi} \operatorname{sen} \phi \operatorname{sen} \theta, \quad \dot{z}=0$, obtenemos $\dot{\psi}+\dot{\phi}\left[\frac{b}{a} \operatorname{sen} \theta+\cos \theta\right]=0 \Rightarrow \psi=-\frac{\sqrt{a^2+b^2}}{a} \phi$
	\item<4-> Donde hemos sustituido el valor de $\theta$ que implica $\operatorname{sen} \theta=b / \sqrt{a^2+b^2}$ y $\cos \theta=a / \sqrt{a^2+b^2}$.
\end{itemize}
}
%%%%% Diapo 8
\subsection{Energías Cinética, Potencial y el Lagrangeano}
\frame{
\frametitle{Energías Cinética, Potencial y el Lagrangeano}
\begin{itemize}  
	\item<1-> Como siempre la energía cinética se construye como 
	$T= T_{tras} + T_{rot} = \frac{1}{2} M(\dot{x}^2+\dot{y}^2+\dot{z}^2)
	+\frac{1}{2}( I_{11} \tilde{\Omega}_1^2 +I_{22} \tilde{\Omega}_2^2 +I_{33} \tilde{\Omega}_3^2 )$
	\item<2-> Los momentos de inercia son: $I_{11} = I_{22}=\frac{1}{4} M a^2$ y $I_{33}=\frac{1}{2} M a^2$.
	\item<3-> Las componentes de la velocidad angular son 
	$\tilde{\Omega}_1  = \dot{\phi} \operatorname{sen} \, \theta \operatorname{sen} \, \psi+\dot{\theta} \cos \psi$; 
	$\quad \tilde{\Omega}_2  = \dot{\phi} \operatorname{sen} \, \theta \cos \psi-\dot{\theta} \operatorname{sen} \, \psi$; 
	$\quad \tilde{\Omega}_3  = \dot{\psi}+\dot{\phi} \cos \theta$.
	\item<4-> Finalmente la energía cinética queda como $T=\frac{M}{8} \frac{a^2 b^2+6 b^4}{a^2+b^2} \dot{\phi}^2$.
	\item<5-> En general la energía potencial la derivamos de la única fuerza posible: la gravitacional ${\bf F} = -Mg(\operatorname{sen} \alpha {\bf \hat{y}} + \cos \alpha {\bf \hat{z}} ) = -{\bf \nabla} V $
	\item<6-> La energía potencial será $V = Mg(y \operatorname{sen} \alpha +z \cos \alpha ) \equiv Mgb(-\cos \phi \operatorname{sen} \theta \operatorname{sen} \alpha +\cos \theta \cos \alpha )$
	\item<7-> La ecuación de movimiento será $\ddot{\phi}=-g \operatorname{sen} \alpha \frac{4\left(a^2+b^2\right)^{3 / 2}}{a^2+6 b^2} \operatorname{sen} \phi$
\end{itemize}
}

\subsection{Rodar sin deslizar y energía cinética}
\frame{
\frametitle{Rodar sin deslizar y energía cinética}
\begin{itemize}
	\item<1-> La ligadura de rodar sin deslizar implica $\dot{{\bf r}}_{op} = 0=\dot{{\bf R}}+  \tilde{ {\bf \Omega}} \times {\bf r}_{cp}$.  
	\item<2->  Por su parte, $\tilde{{\bf \Omega}} \times {\bf r}_{cp} = 
	\left|
	\begin{array}{ccc} 
	{\bf \hat{x}}_1 & {\bf \hat{x}}_2 & {\bf \hat{x}}_3 \\
	\tilde{\Omega}^1 & \tilde{\Omega}^2 & {\tilde \Omega}^3 \\
	0 & -b & 0
	\end{array}\right| = b(\tilde{\Omega}^3 \, {\bf \hat{x}}_1 -\tilde{\Omega}^1\, {\bf \hat{x}}_3) $
	\item<3->  Al proyectar la ecuación de ligadura respecto a la base del Sistema laboratorio tendremos 
	$\dot{x} + b(\tilde{\Omega}^3 ({\bf \hat{x}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{x}} \cdot {\bf \hat{x}}_3))=0 $; $\quad \dot{y} + b(\tilde{\Omega}^3  ({\bf \hat{y}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{y}} \cdot {\bf \hat{x}}_3)) =0$;  $\quad a(\tilde{\Omega}^3  ({\bf \hat{z}} \cdot {\bf \hat{x}}_1) -\tilde{\Omega}^1 ({\bf \hat{z}} \cdot {\bf \hat{x}}_3)) =0 $.
	\item<4-> Por otro lado para el centro de masa \\
	$\tilde{\Omega}^1  = \dot{\phi} \operatorname{sen} \, \theta \operatorname{sen} \, \psi $ 
	$\quad \tilde{\Omega}^2  = \dot{\phi} \operatorname{sen} \, \theta \cos \psi $ 
	$\quad \tilde{\Omega}^3  = \dot{\psi}+\dot{\phi} \cos \theta$
	\item<5-> Entonces, rodar sin deslizar se convierte en $\dot{x}+b \dot{\phi} \cos \phi \cos \theta+b \dot{\psi} \cos \phi  =0$  $\quad \dot{y}+b \dot{\phi} \operatorname{sen} \phi \cos \theta+b \dot{\psi} \operatorname{sen} \phi =0$
	\item<6-> Finalmente la energía cinética es 
	$T = T_{tras} + T_{rot} =\frac{1}{2} M\left(\dot{x}^2+\dot{y}^2\right)+\frac{I_{11}}{2} \left[(\Omega^{1})^2+(\Omega^{2})^2\right]+\frac{I_{33}}{2}  (\Omega^{3})^2$ \\
	$T=\frac{1}{2} M\left(\dot{x}^2+\dot{y}^2\right)+\frac{I_{11}}{2}  \dot{\phi}^2 \cos ^2 \alpha+\frac{I_{33}}{2} (\dot{\phi} \operatorname{sen} \alpha+\dot{\psi})^2$
\end{itemize}
}

		\end{itemize}

Es decir: $T=\frac{1}{2} M\left(\dot{x}^2+\dot{y}^2+\dot{z}^2\right)$ $+\frac{1}{8} M a^2\left[\dot{\phi}^2 \operatorname{sen}^2 \theta+\dot{\theta}^2+2\left(\dot{\phi} \cos \theta^2+\dot{\psi}\right)^2\right]$

			\item $p_{\psi} = \frac{\partial \mathcal{L}}{\partial \dot{\psi}} = {\rm cte} \quad \Rightarrow$

  		\item La ecuación de movimiento será $\ddot{\phi}=-g \operatorname{sen} \alpha \frac{4\left(a^2+b^2\right)^{3 / 2}}{a^2+6 b^2} \operatorname{sen} \phi$
	\end{itemize}

\begin{frame}{Moneda que rueda sin deslizar}
\begin{itemize}
  \item Disco homogéneo de radio \( a \), masa \( M \), seis grados de libertad.
  \item Ligadura: la velocidad del punto de contacto es cero:
  \[
  \dot{\mathbf{r}}_{\text{op}} = \dot{\mathbf{R}} + \boldsymbol{\Omega} \times \mathbf{r}_{\text{cp}} = 0
  \]
  \item Se deducen relaciones entre \( \dot{x}, \dot{y}, \dot{z} \) y \( \dot{\phi}, \dot{\theta}, \dot{\psi} \).
  \item Se toma \( \psi = 0 \) por simetría.
\end{itemize}
\end{frame}

\begin{frame}{Energía y Lagrangiano de la moneda}
\begin{itemize}
  \item Energía cinética:
  \[
  T = \frac{1}{2} M (\dot{x}^2 + \dot{y}^2 + \dot{z}^2) + \frac{1}{2} \boldsymbol{\Omega}^T I \boldsymbol{\Omega}
  \]
  \item Incorporando ligaduras:
  \[
  T = \frac{1}{32} M a^2 \left[ \text{expresión en } \dot{\phi}, \dot{\theta}, \dot{\psi} \right]
  \]
  \item Energía potencial: \( V = mga \operatorname{sen} \theta \)
  \item Lagrangiano: \( \mathcal{L} = T - V \)
\end{itemize}
\end{frame}

\begin{frame}{Simetrías y constantes de movimiento}
\begin{itemize}
  \item Coordenadas cíclicas: \( \phi, \psi \)
  \item Conservación de los momentos conjugados \( p_\phi, p_\psi \)
  \item Ecuación de movimiento para \( \theta \):
  \[
  10\ddot{\theta} + \dots + \frac{8g}{a} \cos\theta = 0
  \]
\end{itemize}
\end{frame}

%---------------------------------------------
\begin{frame}{Moneda en plano inclinado}
\begin{itemize}
  \item Disco unido a varilla sin masa, rueda sobre plano inclinado.
  \item Coordenadas: \( \phi, \theta \); ligaduras reducen el sistema.
  \item Se obtiene:
  \[
  \dot{\psi} + \dot{\phi}\left( \frac{b}{a\operatorname{sen}\theta} + \cos\theta \right) = 0
  \]
  \item Energía cinética: función de \( \dot{\phi} \)
  \item Energía potencial: \( V = Mgb(\ldots) \)
  \item Ecuación de movimiento: 
  \[
  \ddot{\phi} = -\omega^2 \operatorname{sen}\phi
  \]
\end{itemize}
\end{frame}

\begin{frame}{Cono que rueda sin deslizar}
\begin{itemize}
  \item Cono de altura \( h \), masa \( m \), ángulo de vértice \( \alpha \).
  \item Ligaduras:
  \[
  x = \frac{3h}{4} \cos\phi \operatorname{sen}\theta,\quad y = \frac{3h}{4} \operatorname{sen}\phi \operatorname{sen}\theta
  \]
  \item Rotación sin deslizamiento: impone condiciones sobre \( \dot{x}, \dot{y} \)
  \item Energía cinética:
  \[
  T = \frac{1}{2} M (\dot{x}^2 + \dot{y}^2) + \frac{1}{2} I_{11} \dot{\phi}^2 \cos^2\alpha + \frac{1}{2} I_{33} (\dot{\phi} \operatorname{sen}\alpha + \dot{\psi})^2
  \]
\end{itemize}
\end{frame}

\begin{frame}{Conclusiones}
\begin{itemize}
  \item La rodadura sin deslizamiento impone ligaduras no integrables.
  \item El tratamiento lagrangiano permite incorporar las simetrías del problema.
  \item Casos como la moneda, plano inclinado y cono son útiles para ilustrar ligaduras no holónomas.
\end{itemize}
