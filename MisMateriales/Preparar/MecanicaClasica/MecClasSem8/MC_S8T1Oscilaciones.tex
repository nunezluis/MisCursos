\frame{
  \frametitle{Pequeñas oscilaciones 1D 1/2}
   \begin{itemize}  
  	\item<1-> Como vimos en la clase de estabiliad dado un $\mathcal{L}=\frac{1}{2} c\dot{q}^2-V_{\mathrm{ef}}(q)$
	\item<2-> La ecuación de movimiento es $c \ddot{q}=f_{\mathrm{ef}}(q) \equiv -\frac{\partial V_{\mathrm{ef}}}{\partial q}$
	\item<3-> Que tendrá un mínimo en $q_0$, cuando  $f_{\mathrm{ef}}\left(q_0\right)=\left.0 \Rightarrow \frac{\partial V_{\mathrm{ef}}}{\partial q}\right|_{q_0}=0$
	\item<4-> Será estable si $\left.\frac{\partial V_{\mathrm{ef}}^2}{\partial q^2}\right|_{q_0}>0$
	\item<5-> Igual que en el caso anterior perturbamos alrededor del mínimo
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/MinimoOscila.jpg}
   	\end{figure}
	\item<6-> Desarrollamos por Taylor, $V_{\text {ef }}(q)$ alrededor de $q=q_0$, y tenemos  $V_{\mathrm{ef}}(q)=V_{\mathrm{ef}}\left(q_0+\eta\right)=V\left(q_0\right)+ \cancelto{0}{\left.\frac{\partial V_{\mathrm{ef}}}{\partial q}\right|_{q_0}} \eta+\left.\frac{1}{2} \frac{\partial^2 V_{\mathrm{ef}}}{\partial q^2}\right|_{q_0} \eta^2+\cdots,$
    \end{itemize}
}

  	\item<1-> Despreciando términos en potencias de $\eta$ de orden superior al cuadrático, tenemos $V_{\mathrm{ef}}(q)=V_{\mathrm{ef}}\left(q_0+\eta\right) \approx \frac{1}{2} K\left(q-q_0\right)^2=\frac{1}{2} K \eta^2$
	\item<2-> La ecuación de movimiento será $c \ddot{\eta} =-\frac{\partial V_{\text {ef }}}{\partial q}=-K\left(q-q_0\right) \equiv -K \eta$
	\item<3-> Entonces, $\ddot{\eta}+\omega^2 \eta=0$, donde $\omega^2 \equiv \frac{K}{c}=\left.\frac{1}{c} \frac{\partial^2 V_{\mathrm{ef}}}{\partial q^2}\right|_{q_0}$ es la frecuencia angular de las pequeñas oscilaciones alrededor de $q_0$.
	\item<4-> Que tendrá como solución \\ $\eta(t)=c_1 \cos \omega t+c_2 \operatorname{sen} \omega t=A \cos (\omega t+\varphi) \equiv \operatorname{Re}\left[A e^{i(\omega t+\varphi)}\right]=\operatorname{Re}\left(a e^{i \omega t}\right)$ donde $a=A e^{i \varphi}$ es la amplitud compleja

   \end{itemize}

	\item<2-> La configuración de equilibrio del sistema en $\left\{q_{0 i}: i=1, \ldots, s\right\}$, con $\left.\frac{\partial V}{\partial q_i}\right|_{q_{0 i}}=0, \quad i=1,2, \ldots, s$

   	\end{figure}
	\item<3-> Perturbando las $q_i$, tendremos $q_i=q_{0 i}+\eta_i,$  con $\eta_i \rightarrow 0 \quad \left(\frac{\eta_i}{q_{0 i}} \ll 1\right)$
	\item<4-> El valor del potencial $V\left(q_1, \ldots, q_s\right)$ cerca de la configuración de equilibrio se obtiene de la expansión de Taylor en varias variables de $V\left(q_1, \ldots, q_s\right)$ alrededor de $\left\{q_{0 i}\right\}$, con $q_i=q_{0 i}+\eta_i$,
   \end{itemize}

\frame{
  \frametitle{Oscilaciones con varios grados de libertad 2/3}
  \begin{itemize}  
  	\item<1-> Esto es 
	$V\left(q_1, \ldots, q_s\right)=V\left(q_{01}, \ldots, q_{0 s}\right)+ \cancelto{0}{\sum_i\left(\frac{\partial V}{\partial q_i}\right)_{\left\{q_{0 i}\right\}}} \eta_i +\frac{1}{2} \sum_i \sum_j\left(\frac{\partial^2 V}{\partial q_i \partial q_j}\right)_{\left\{q_{0 i}\right\}} \eta_i \eta_j+\cdots$ 
	\item<2-> Los $V\left(q_{01}, \ldots, q_{0 s}\right)$ es un valor constante y las derivadas parciales están evaluadas en $\left\{q_{0 i}\right\}=\left(q_{01}, \ldots, q_{0 s}\right)$.
	\item<3-> En la configuración de equilibrio $q_i=q_{0 i}+\eta_i$, el potencial se puede expresar como $V\left(q_1, \ldots, q_s\right)=V\left(\eta_1, \ldots, \eta_s\right)=\frac{1}{2} \sum_{i, j} V_{i j} \eta_i \eta_j$
	\item<4-> donde los coeficientes $V_{i j} \equiv\left(\frac{\partial^2 V}{\partial q_i \partial q_j}\right)_{\left\{q_{0 i}\right\}}$ son simétricos, $V_{i j}=V_{j i}$
	\item<5-> La energía cinética del sistema es $T=\frac{1}{2} \sum_{i, j} T_{i j} \dot{q}_i \dot{q}_j$ donde los coeficientes $T_{i j}=T_{j i}$ representan parámetros constantes que dependen de propiedades del sistema (masas, longitudes, etc).
	\item<6-> Para pequeños desplazamientos $T=\frac{1}{2} \sum_{i, j} T_{i j} \dot{\eta}_i \dot{\eta}_j$
	\item<7-> El Lagrangiano del sistema cerca de la configuración de equilibrio es $\mathcal{L}=T-V=\frac{1}{2} \sum_{i, j}\left(T_{i j} \dot{\eta}_i \dot{\eta}_j-V_{i j} \eta_i \eta_j\right) \quad i, j=1,2, \ldots, s$

   \end{itemize}
}

\section{Sección}
\frame{
  \frametitle{Oscilaciones con varios grados de libertad 3/3}
  \begin{itemize}  
  	\item<1->  La ecuación de Lagrange en $\eta_k$ es $\frac{d}{d t}\left(\frac{\partial L}{\partial \dot{\eta}_k}\right)-\frac{\partial L}{\partial \eta_k}=0$
	\item<2-> $\frac{\partial L}{\partial \dot{\eta}_k}  =\frac{1}{2} \sum_{i, j} T_{i j}\left(\delta_{i k} \dot{\eta}_j+\delta_{j k} \dot{\eta}_i\right)=\frac{1}{2}\left(\sum_j T_{k j} \dot{\eta}_j+\sum_i T_{i k} \dot{\eta}_i\right)=\sum_j T_{k j} \dot{\eta}_j $
	\item<3-> $\frac{\partial L}{\partial \eta_k}  =-\frac{1}{2} \sum_{i, j}\left(V_{i j} \delta_{i k} \eta_j+V_{i j} \delta_{j k} \eta_i\right)=-\sum_j V_{k j} \eta_j$
	\item<4-> la ecuación de movimiento para $\eta_k$ queda $\sum_j\left(T_{k j} \ddot{\eta}_j+V_{k j} \eta_j\right)=0$, donde cada término de la suma tiene la forma de una ecuación para un oscilador armónico.
	\item<5-> Como la solución es $\eta_j(t)=a_j e^{i \omega t}$ tenemos
	$ \sum_j\left(-\omega^2 T_{k j}+V_{k j}\right) a_j e^{i \omega t}=0  \quad \Rightarrow \sum_n \left(V_{m n}-\omega^2 T_{m n }\right) a_n=0$
	\item<6-> Es decir, para dos grados de libertad $s= 1, 2$ tendremos \\ 
	$m=1: \left(V_{11}-\omega^2 T_{11}\right) a_1+\left(V_{12}-\omega^2 T_{12}\right) a_2=0$ \\
	$m=2:  \left(V_{21}-\omega^2 T_{21}\right) a_1+\left(V_{22}-\omega^2 T_{22}\right) a_2=0$
	\item<7-> En general, existe solución no trivial $\eta_j(t) \neq 0$ si $a_j \neq 0, \forall j$, cuando $\operatorname{det}\left|V_{i j}-\omega^2 T_{i j}\right|=0$. Las frecuencias características $\omega$ deben ser reales para que las soluciones tengan sentido físico.
   \end{itemize}
}

\section{Ejemplo: Osciladores Acoplados}
\frame{
  \frametitle{Osciladores Acoplados 1/2}
  \begin{itemize}  
  	\item<1-> Encontrar las frecuencias de oscilación para un sistema de dos partículas de masa $m$ conectadas com resortes horizontales, de constantes $k$ y longitud en reposo $l$. 
	\begin{figure}[t]
		\includegraphics[width=1.5in]{Figuras/OsciladoresAcoplados.png}
   	\end{figure}
	 \item<2-> El sistema tiene dos grados de libertad $(s=2)$. Para pequeños desplazamientos $\eta_1$ y $\eta_2$, con $x_i=x_{0 i}+\eta_i$,
	 \item<3-> La energía cinética para pequeños desplazamientos del equilibrio es $T=\frac{1}{2} m \dot{\eta}_1^2+\frac{1}{2} m \dot{\eta}_2^2=\frac{1}{2} \sum_{i, j} T_{i j} \dot{\eta}_i \dot{\eta}_j$, con $T_{11}=m, \quad T_{22}=m, \quad T_{12}=T_{21}=0$
	 \item<4-> La energía potencial del sistema para pequeños desplazamientos es $V=\frac{1}{2} k \eta_1^2+\frac{1}{2} k\left(l^{\prime}-l\right)^2+\frac{1}{2} k \eta_2^2=\frac{1}{2} k\left[\eta_1^2+\left(\eta_2-\eta_1\right)^2+\eta_2^2\right]$
	 \item<5-> Donde $l^{\prime}-l=\left(x_2-x_1\right)-\left(x_{02}-x_{01}\right)=\eta_2-\eta_1$
   \end{itemize}
}

\frame{
  \frametitle{Osciladores Acoplados 2/2}
  \begin{itemize}  
  	\item<1-> Entonces $V=\frac{1}{2}\left[2 k \eta_1^2+2 k \eta_2^2-2 k \eta_1 \eta_2\right]=\frac{1}{2} \sum_{i, j} V_{i j} \eta_i \eta_j$ con $V_{11}=2 k, \quad V_{22}=2 k, \quad V_{12}=V_{21}=-k$.
	\item<2-> Por lo tanto $
	\left|\begin{array}{ll}
	V_{11}-\omega^2 T_{11} & V_{12}-\omega^2 T_{12} \\
	V_{21}-\omega^2 T_{21} & V_{22}-\omega^2 T_{22}
	\end{array}\right|=0$
	\item<3-> Es decir
	$\left|\begin{array}{cc}
	2 k-\omega^2 m & -k \\
	-k & 2 k-\omega^2 m
	\end{array}\right|=0$
	\item<4-> La ecuación característica resultante es 
	$\left(2 k-\omega^2 m\right)^2-k^2=0 \Rightarrow 2 k-\omega^2 m= \pm k \Rightarrow \omega^2=\frac{2 k \pm k}{m}$
	\item<5-> Finalmente 
	$\omega_1=\sqrt{\frac{3 k}{m}}, \quad \omega_2=\sqrt{\frac{k}{m}}$
   \end{itemize}
}
