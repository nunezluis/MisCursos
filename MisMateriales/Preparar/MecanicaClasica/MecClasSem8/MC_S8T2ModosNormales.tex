\section{Repasamos los Modos normales de oscilación}
\frame{
  \frametitle{Modos normales de oscilación 1/2}
   \begin{itemize}  
  	\item<1-> Las $s$ ecuaciones de movimiento para un sistema con pequeñas oscilaciones $\left\{\eta_1, \ldots, \eta_s\right\}$ alrededor del equilibrio de $\left\{q_{01},  \ldots, q_{0 s}\right\}$ son $\sum_j\left(T_{i j} \ddot{\eta}_j+V_{i j} \eta_j\right)=0, \quad i=1,2, \ldots, s$
	\item<2-> Si suponemos una solución de la forma $\eta_j(t)=a_j e^{i \omega t}$ tendremos $\sum_j\left(V_{i j}-\omega^2 T_{i j}\right) a_j=0 \quad i=1,2, \ldots, s$
	\item<3-> La condición $\operatorname{det}\left|V_{i j}-\omega^2 T_{i j}\right|=0 $ es decir $\left|\begin{array}{ccc}V_{11}-\omega^2 T_{11} & V_{12}-\omega^2 T_{12} & \ldots \\ V_{21}-\omega^2 T_{21} & V_{22}-\omega^2 T_{22} & \\ V_{31}-\omega^2 T_{31} & \\ \vdots & & \end{array}\right|=0$
	\item<4-> Esto permite calcular las $s$ frecuencias de pequeñas oscilaciones $\omega_n, \quad n=1,2, \ldots, s$ como soluciones al polinomio característico
\end{itemize}
}
%%%%% Diapo 2
%\section{Sección}
\frame{
  \frametitle{Modos normales de oscilación 2/2}
\begin{itemize}
	\item<1-> 	Para cada $\omega_n$, existe un sistema de $s$ ecuaciones para $a_j\left(\omega_n\right)$. \\ 
 	Si  $s=2$ \\
	$i=1:  \left(V_{11}-\omega_n^2 T_{11}\right) a_1+\left(V_{12}-\omega_n^2 T_{12}\right) a_2=0$ \\
	$i=2:  \left(V_{21}-\omega_n^2 T_{21}\right) a_1+\left(V_{22}-\omega_n^2 T_{22}\right) a_1=0$ \\ Para cada $\omega_n$ tendremos $2$ ecuaciones lineales para $a_1\left(\omega_n\right)$ y $a_2\left(\omega_n\right)$
	\item<3-> La solución general, $\eta_j(t)$, será la superposición de las soluciones $\eta_j(t)=\sum_n c_n a_j\left(\omega_n\right) e^{i \omega_n t}$, donde $c_n$ son las fases complejas.
	\item<4-> Si  $\xi_n \equiv c_n e^{i \omega_n t}, \quad n=1,2, \ldots, s$, tendremos $\eta_j(t)=\sum_n a_j\left(\omega_n\right) \xi_n$ la solución es una combinación lineal de las coordenadas normales
	\item<5-> Cada coordenada normal $\xi_n$ satisface la ecuación $\ddot{\xi}_n+\omega_n^2 \xi_n=0$.
	\item<6-> En el caso de $s=2$, las soluciones generales para los pequeños desplazamientos son $\eta_1=a_1\left(\omega_1\right) \xi_1+a_1\left(\omega_2\right) \xi_2, \quad \eta_2=a_2\left(\omega_1\right) \xi_1+a_2\left(\omega_2\right) \xi_2$
	\item<7-> Para $\xi_2$ tenemos $\eta_1=a_1\left(\omega_2\right) \xi_2, \quad \eta_2=a_2\left(\omega_2\right) \xi_2
$,  2 pequeños desplazamientos que oscilas con la frecuencia $\omega_2$ alrededor de su posición de equilibrio con amplitudes $a_1\left(\omega_2\right)$ y $a_2\left(\omega_2\right)$.
\end{itemize}
}

\section{Osciladores Acoplados}
\subsection{Los osciladores}
\frame{
  \frametitle{Los Osciladores}
Encontrar las frecuencias de oscilación para un sistema de dos partículas de masa $m$ conectadas com resortes horizontales, de constantes $k$ y longitud en reposo $l$. 
	\begin{figure}[t]
		\includegraphics[width=1.4in]{Figuras/OsciladoresAcoplados.png}
   	\end{figure}  
  \begin{itemize}  
	 \item<1-> El sistema tiene dos grados de libertad $(s=2)$. Para pequeños desplazamientos $\eta_1$ y $\eta_2$, con $x_i=x_{0 i}+\eta_i$,
	 \item<2-> La energía cinética para pequeños desplazamientos del equilibrio es $T=\frac{1}{2} m \dot{\eta}_1^2+\frac{1}{2} m \dot{\eta}_2^2=\frac{1}{2} \sum_{i, j} T_{i j} \dot{\eta}_i \dot{\eta}_j$, con $T_{11}=m, \quad T_{22}=m, \quad T_{12}=T_{21}=0$
	 \item<3-> La energía potencial del sistema para pequeños desplazamientos es $V=\frac{1}{2} k \eta_1^2+\frac{1}{2} k\left(l^{\prime}-l\right)^2+\frac{1}{2} k \eta_2^2=\frac{1}{2} k\left[\eta_1^2+\left(\eta_2-\eta_1\right)^2+\eta_2^2\right]$
	 \item<4-> Donde $l^{\prime}-l=\left(x_2-x_1\right)-\left(x_{02}-x_{01}\right)=\eta_2-\eta_1$
   \end{itemize}
}
%%%%% Diapo 11
\subsection{Las ecuaciones de movimiento y sus modos nomales}
\frame{
  \frametitle{Ecuaciones de movimiento y sus modos nomales}
  \begin{itemize}  
	\item<1-> $V=\frac{1}{2}\left[2 k \eta_1^2+2 k \eta_2^2-2 k \eta_1 \eta_2\right]=\frac{1}{2} \sum_{i, j} V_{i j} \eta_i \eta_j$ con $V_{11}=2 k, \quad V_{22}=2 k, \quad V_{12}=V_{21}=-k$.
	\item<3-> $\sum_j\left(V_{i j}-\omega^2 T_{i j}\right) a_j=0 \Rightarrow 
	\operatorname{det}
	\begin{vmatrix}
		k\begin{pmatrix}
		2 & -1 \\
		-1  & 2
		\end{pmatrix}
		-m\omega^2
	\begin{pmatrix}
		1 & 0 \\
		0 & 1
		\end{pmatrix}
	\end{vmatrix} = 0
	$ 
	\item<3-> Es decir
	$\operatorname{det}\begin{vmatrix}
	 \frac{2k}{m} -\omega^2 & -\frac{k}{m} \\
	-\frac{k}{m} &  \frac{2k}{m} -\omega^2
	\end{vmatrix}=0 \Rightarrow \left(2 k-\omega^2 m\right)^2-k^2=0$
	\item<4-> La ecuación característica es 
	$2 k-\omega^2 m= \pm k \Rightarrow \omega_{\pm}^2=\frac{2 k \pm k}{m}$
	\item<5-> La ecuación matricial es 
	$\begin{pmatrix}
		2 & -1 \\
		-1  & 2
		\end{pmatrix}
		\begin{pmatrix}
		a^1_{\pm} \\
		a^2_{\pm}
		\end{pmatrix}
		= \frac{m \omega_{\pm}^2}{k}
	\begin{pmatrix}
		a^1_{\pm} \\
		a^2_{\pm}
		\end{pmatrix}$
	\item<6-> Claramente una ecuación de autovalores
	\item<7-> Entonces, las ecuaciones serán  
	$\left\{ \begin{matrix}
	2a^1_{\pm} -a^2_{\pm}= & \frac{m \omega_{\pm}^2}{k} a^1_{\pm} \\ \\
	-a^1_{\pm} +2a^2_{\pm}= & \frac{m \omega_{\pm}^2}{k} a^1_{\pm}
	\end{matrix}
	\right.$ 
   \end{itemize}
}
%%%%% Diapo 2
\subsection{De las frecuencias $\omega_i$ a las amplitudes $a^i$}
\frame{
  \frametitle{De las frecuencias $\omega_i$ a las amplitudes $a^i$}
  \begin{itemize}  
  	\item<1-> Si definimos $\lambda_\pm := \frac{m\omega_{\pm}^2}{k}$, entonces 
	$\left\{ \begin{matrix}
	2a^1_\pm - a^2_\pm = \lambda_\pm a^1_\pm  \\ \\
	-a^1_\pm + 2a^2_\pm = \lambda_\pm a^2_\pm 
	\end{matrix} \right.$
	\item<2-> Entonces $(2 - \lambda_\pm)a^1_\pm - a^2_\pm = 0 \quad$ y $\quad -a^1_\pm + (2 - \lambda_\pm)a^2_\pm = 0$
	\item<3-> Al despejar $a^2_\pm$ de la primera y sustituirla en la segunda, tendremos $a^1_\pm \left[(2 - \lambda_\pm)^2 - 1\right] = 0 \quad \Rightarrow (2 - \lambda_\pm)^2 = 1 \quad \Rightarrow \quad 2 - \lambda_\pm = \pm 1$
	\item<4-> Entonces, si $2 - \lambda_\pm = -1 \Rightarrow \lambda_+ = 3$  y  si $2 - \lambda_\pm = 1 \Rightarrow \lambda_- = 1$, que son los dos valores que corresponden a dos valores que optuvimos de la ecuación característica $\omega_{\pm} = \frac{2 k \pm k}{m} \Rightarrow \omega_{+} = \frac{3 k}{m}$ y $\omega_{-} = \frac{ k}{m}$
	\item<5-> Finalmente, $\frac{a^2_\pm}{a^1_\pm} = (2 - \lambda_\pm) \Rightarrow 
	\left\{\ 
	\begin{matrix}
		\frac{a^2_+}{a^1_+} = +1 & \text{en fase} \\ \\ 
		\frac{a^2_-}{a^1_-} = -1 & \text{en contra fase}
	\end{matrix}
	 \right.$
   \end{itemize}
}

\section{Los Modos Normales Oscilación para el CO2}
\subsection{Descripción del sistema CO2}
\frame{
\frametitle{Descripción del sistema CO2}
\begin{itemize}  
	\item<1-> Consideremos el siguiente sistema que representa la molécula de CO2
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/OscilaCO2.png}
   	\end{figure}
	$M$ masa del átomo $\mathrm{C}; m$ masa de los átomos $\mathrm{O}$; $l$ la separación entre posiciones de equilibrio; la constante elástica $k$ de interacción C-O; $l_1$, $l_2$, las distancias de los átomos fuera del equilibrio.
	\item<2-> Sean $x_{01}, x_{02}, x_{03}$ las posiciones de equilibrio de las tres partículas y $\eta_i=x_i-x_{0 i}$, con $i=1,2,3$ los desplazamientos del equilibrio.
\end{itemize}
}
%%%%% Diapo 2
\subsection{Energías cinética y potencial}
\frame{
\frametitle{Energías cinética y potencial}
\begin{itemize}
	\item<1-> La energía cinética es $T = \frac{1}{2} m \dot{\eta}_1^2+\frac{1}{2} M \dot{\eta}_2^2+\frac{1}{2} m \dot{\eta}_3^2$
	\item<2-> La energía potencial es $V=\frac{1}{2} k\left(l_1-l\right)^2+\frac{1}{2} k\left(l_2-l\right)^2$. 
	\item<3-> Como $ l_1-l=\left(x_2-x_1\right)-\left(x_{02}-x_{01}\right)=\eta_2-\eta_1$ y \\  $l_2-l=\left(x_3-x_2\right)-\left(x_{03}-x_{02}\right)=\eta_3-\eta_2$
	\item<4-> Tendremos $V  =\frac{1}{2} k\left(\eta_2-\eta_1\right)^2+\frac{1}{2} k\left(\eta_3-\eta_2\right)^2 \Rightarrow $ \\ $V =\frac{1}{2} k\left(\eta_1^2+2 \eta_2^2+\eta_3^2-2 \eta_1 \eta_2-2 \eta_2 \eta_3\right)$
	\item<4-> Entonces, como $T=\frac{1}{2} \sum_{i, j} T_{i j} \dot{\eta}_i \dot{\eta}_j$, y  $V=\frac{1}{2} \sum_{i, j} V_{i j} \eta_i \eta_j$
	\item<5-> Tendremos 
	$T_{ij} = \left(
	\begin{array}{lll}
	T_{11}=m & T_{12}=0 & T_{13}=0 \\
	T_{21}=0 & T_{22}=M & T_{23}=0 \\
	T_{31}=0 & T_{32}=0 & T_{33}=m
	\end{array}
	\right)$
	\item<6-> y 
	$V_{ij} = \left(
	\begin{array}{lll}
		V_{11}=k & V_{12}=-k & V_{13}=0 \\
		V_{21}=-k & V_{22}=2 k & V_{23}=-k \\
		V_{31}=0 & V_{32}=-k & V_{33}=k
	\end{array} \right)$
\end{itemize}
}
%%%%% Diapo 2
\subsection{Las frecuencias de oscilación $\omega_n$}
\frame{
\frametitle{Las frecuencias de oscilación $\omega_n$}
\begin{itemize}  
	\item<1-> La condición det $\left|V_{i j}-\omega^2 T_{i j}\right|=0$, implica
	$\left|\begin{array}{ccc}
		k-\omega^2 m & -k & 0 \\
		-k & 2 k-\omega^2 M & -k \\
		0 & -k & k-\omega^2 m
	\end{array}\right|=0$ para las frecuencias $\omega_n$
	\item<2-> La ecuación característica cúbica para $\omega_n$, es $\left(k-\omega^2 m\right)\left[\left(2 k-\omega^2 M\right)\left(k-\omega^2 m\right)-k^2\right]-k^2\left(k-\omega^2 m\right)=0$  \\ $\Rightarrow \quad \omega^2\left(k-\omega^2 m\right)\left[k(M+2 m)-\omega^2 M m\right]=0$
	\item<3-> Con soluciones $\omega_1=0, \quad \omega_2=\sqrt{\frac{k}{m}}, \quad \omega_3=\sqrt{\frac{k}{m}\left(1+\frac{2 m}{M}\right)}$ 
\end{itemize}
}
%%%%% Diapo 2
\subsection{Ecuaciones generales y modo normal de oscilación para $\omega_1=0$}
\frame{
\frametitle{Modo normal de oscilación para $\omega_1=0$}
\begin{itemize}  
	\item<1-> Las amplitudes $a_i$ surgen de las 3 ecuaciones para cada $\omega_n$,
	$\begin{array}{cccc}
		i=1: & \left(k-\omega_n^2 m\right) a_1-k a_2 & = & 0 \\
		i=2: & -k a_1+\left(2 k-\omega_n^2 M\right) a_2-k a_3 & =0 \\
		i=3: & -k a_2+\left(k-\omega_n^2 m\right) a_3 & = & 0
		\end{array}$
	\item<2-> La frecuencia angular $\omega_1=0$ es una traslación uniforme de la molécula ya que 
	$\ddot{\zeta}_1=0 \Rightarrow \dot{\zeta}_1=\text { cte } \Rightarrow$ reposo o velocidad constante
	\item<3-> Entonces para $\omega_1=0$, tenemos  $a_1(\omega_1)=a_2(\omega_1)=a_3(\omega_1)$
	\begin{figure}[t]
		\includegraphics[width=1.8in]{Figuras/OscilaCO2Modo1.png}
   	\end{figure}
\end{itemize}
}

\section{Recapitulando}
\frame{
  \frametitle{Recapitulando}
  \begin{itemize}  
  	\item<1-> Formulación general de pequeñas oscilaciones:  $ \sum_j (T_{ij} \ddot{\eta}_j + V_{ij} \eta_j) = 0$
	\item<2-> Para soluciones \( \eta_j(t) = a_j e^{i\omega t} \), obtenemos:  $\sum_j (V_{ij} - \omega^2 T_{ij}) a_j = 0 $
	\item<3-> Con soluciones no triviales a partir de $\det |V_{ij} - \omega^2 T_{ij}| = 0$
	\item<4-> Se obtiene los autovalores \( \omega_n \) y autovectores \( a_j(\omega_n) \)
	\item<5-> Las soluciones generales son: $ \eta_j(t) = \sum_n a_j(\omega_n) \xi_n(t), \quad \ddot{\xi}_n + \omega_n^2 \xi_n = 0$
	\item<6-> Dos Osciladores Acoplados
	\begin{itemize}
	\item \( T = \frac{1}{2}m(\dot{\eta}_1^2 + \dot{\eta}_2^2) \),  y  \( V = \frac{1}{2}k[\eta_1^2 + (\eta_2 - \eta_1)^2 + \eta_2^2] \)
	\item con las frecuencias normales:  $\omega_1 = \sqrt{\frac{k}{m}}, \quad \omega_2 = \sqrt{\frac{3k}{m}}$
	\end{itemize}
	\item<7-> La molécula de $CO_2$
	\begin{itemize}
	\item Sistema lineal O?C?O: 3 masas, 2 resortes.
	\item Tres modos normales: \( \omega_1 = 0 \): traslación;  \( \omega_2 = \sqrt{\frac{k}{m}} \): O--C en oposición de fase;  \( \omega_3 = \sqrt{\frac{k}{m}(1 + 2m/M)} \): O--C en fase.
	\end{itemize}
   \end{itemize}
}
