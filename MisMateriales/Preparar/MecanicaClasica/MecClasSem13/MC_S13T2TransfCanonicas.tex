\section{Transformaciones Puntuales}
\frame{
  \frametitle{Transformaciones Puntuales}
   \begin{itemize}  
  	\item<1-> Las posiciones en el espacio pueden ser descritas por diferentes sistemas de coordenadas $\left\{q_i\right\}$: cartesianas, esféricas, cilíndricas, etc.
	\item<2-> La forma de las ecuaciones de Lagrange no depende del conjunto de coordenadas $\left\{q_i\right\} \leftrightarrow \left\{Q_i\right\}$: 
	$\frac{d}{d t}\left(\frac{\partial L}{\partial \dot{q}_i}\right)-\frac{\partial L}{\partial q_i}=0 \leftrightarrow  \frac{d}{d t}\left(\frac{\partial L}{\partial \dot{Q}_i}\right)-\frac{\partial L}{\partial Q_i}=0$ 
	\item<3-> En el espacio de fase podemos realizar una {\bf transformación puntual de coordenadas} 
	$\left\{q_i, p_i\right\} \leftrightarrow \left\{Q_i, P_i\right\}\Rightarrow \mathcal{H}\left(q_i, p_i, t\right) \leftrightarrow \tilde{\mathcal{H}}\left(Q_i, P_i, t\right)$
	\item<4-> En general, la forma de las ecuaciones de Hamilton no se preserva en las nuevas coordenadas y momentos $\left\{Q_i, P_i\right\}$.
	\item<5-> Consideremos la siguiente transformacion puntual $\left\{q_i, p_i\right\} \rightarrow\left\{Q_i, P_i\right\}$ en el espacio de fase, $Q_i=-p_i, \quad P_i=q_i$
	\item<6-> Entonces
	$\dot{q}_i=\frac{\partial \mathcal{H}}{\partial p_i} \Rightarrow \dot{P}_i=\frac{\partial \tilde{\mathcal{H}}}{\partial p_i}=\sum_{k=1}^s\left(\frac{\partial \tilde{\mathcal{H}}}{\partial Q_k} \frac{\partial Q_k}{\partial p_i}+\frac{\partial \tilde{\mathcal{H}}}{\partial P_k} \frac{\partial P_k}{\partial p_i}\right) 
=-\sum_k \frac{\partial \tilde{\mathcal{H}}}{\partial Q_k} \delta_{i k}=-\frac{\partial \tilde{\mathcal{H}}}{\partial Q_i}$ \\
$\dot{p}_i=-\frac{\partial \mathcal{H}}{\partial q_i} \Rightarrow \dot{Q}_i=-\frac{\partial \tilde{\mathcal{H}}}{\partial q_i}=-\sum_{k=1}^s\left(\frac{\partial \tilde{\mathcal{H}}}{\partial Q_k} \frac{\partial Q_k}{\partial q_i}+\frac{\partial \tilde{\mathcal{H}}}{\partial P_k} \frac{\partial P_k}{\partial q_i}\right) 
=-\sum_k \frac{\partial \tilde{\mathcal{H}}}{\partial P_k} \delta_{i k}=-\frac{\partial \tilde{\mathcal{H}}}{\partial P_i}$ Claramente no se cumplen
    \end{itemize}
}

\section{Transformaciones Canónicas}
\frame{
\frametitle{Transformaciones Canónicas}
\begin{itemize}  
	\item<1-> Una {\bf transformación canónica} es una transformación puntual que mantiene invariante la forma de las ecuaciones de Hamilton.
	$$
	\begin{array}{r|c|l}
	\mathcal{H}\left(q_i, p_i, t\right) & \rightarrow \text { Transformación canónica }  \rightarrow & \tilde{\mathcal{H}}\left(Q_i, P_i, t\right) \\
	\dot{q}_i=\frac{\partial \mathcal{H}}{\partial p_i} &  P_i=P_i\left(q_j, p_j, t\right)   &  \dot{Q}_i=\frac{\partial \tilde{\mathcal{H}}}{\partial P_i}\\
	\dot{p}_i=-\frac{\partial \mathcal{H}}{\partial q_i} & Q_i\left(q_j, p_j, t\right) &  \dot{P}_i=-\frac{\partial \tilde{\mathcal{H}}}{\partial Q_i}  
	\end{array}
	$$
	\item<2-> Las transformaciones canónicas son particularmente útiles cuando el Hamiltoniano transformado $\tilde{\mathcal{H}}\left(Q_i, P_i, t\right)$ no depende explícitamente de alguna coordenada $Q_j$ o momento $P_j$
\end{itemize}
}

\section{Transformacion Canónica y Principio de Mínima Acción}
\frame{
\frametitle{Principio de Mínima Acción 1/2}
\begin{itemize}  
	\item<1-> La condición para que una transformación $\left\{q_i, p_i\right\} \rightarrow\left\{Q_i, P_i\right\}$ sea canónica se deriva a partir de la equivalencia del Principio de Mínima Acción en ambos conjuntos de variables del espacio de fase.
	\item<2-> El Principio de Mínima Acción para las variables $\left\{q_i, p_i\right\}$, implica $S  =\int_{t_1}^{t_2} \mathcal{L} d t \Rightarrow \delta S=0  \Rightarrow \delta\left(\int_{t_1}^{t_2} \mathcal{L} d t\right)=0$
 	\item<3-> Como el Hamiltoniano $\mathcal{H}\left(q_i, p_i, t\right)=\sum_{i=1}^s p_i \dot{q}_i- \mathcal{L}$
	\item<4-> Tendremos 
	$\delta S=\delta\left(\int_{t_1}^{t_2} \sum_{i=1}^s p_i \dot{q}_i-\mathcal{H}\right) d t=0 \Leftrightarrow$ genera las ecuaciones
	$\dot{q}_i=\frac{\partial \mathcal{H}}{\partial p_i}, \quad \dot{p}_i=-\frac{\partial \mathcal{H}}{\partial q_i}, \quad i=1, \ldots, n$
	\item<5->	Por su parte $ \delta \tilde{S} = \delta\left(\int_{t_1}^{t_2} \sum_{i=1}^s P_i \dot{Q}_i- \tilde{\mathcal{H}}\right) d t=0$ genera $\dot{Q}_i=\frac{\partial \tilde{\mathcal{H}}}{\partial P_i}, \quad \dot{P}_i=-\frac{\partial \tilde{\mathcal{H}}}{\partial Q_i}, \quad i=1, \ldots, n$
\end{itemize}
}

\frame{
\frametitle{Principio de Mínima Acción 2/2}
\begin{itemize}
	\item<1-> Como las variaciones  $\delta S=0$ y  $ \delta \tilde{S} =0$ de forma independiente generan las correspondientes ecuaciones de Hamilton para las variables $\left\{q_i, p_i\right\}$ y $\left\{Q_i, P_i\right\}$ . 
	\item<2-> Ambas formulaciones del Principio de Mínima Acción conducen a ecuaciones equivalentes si los integrandos  difieren, en una derivada total 
	$\sum_{i=1}^s p_i \dot{q}_i-\mathcal{H}=\sum_{i=1}^s P_i \dot{Q}_i- \tilde{\mathcal{H}} +\frac{d  \mathcal{F}}{d t}$
	\item<3-> Con lo cual $\frac{d  \mathcal{F}}{d t}=\sum_{i=1}^s p_i \dot{q}_i-\sum_{i=1}^s P_i \dot{Q}_i+\left(\tilde{\mathcal{H}}-\mathcal{H}\right)$ 
	\item<4-> La función  $\mathcal{F}$ se llama función generadora de la transformación canónica $\left\{q_i, p_i\right\} \rightarrow$ $\left\{Q_i, P_i\right\}$.
\end{itemize}
}

\section{Función Generadora}
\frame{
\frametitle{Función Generadora}
\begin{itemize}  
	\item<1->  Dada una  $\mathcal{F}\left(q_i, p_i, Q_i, P_i, t\right)$, su derivada total $\frac{d  \mathcal{F}}{d t}$ debe satisfacer la condición 
	$\frac{d  \mathcal{F}}{d t}=\sum_{i=1}^s p_i \dot{q}_i-\sum_{i=1}^s P_i \dot{Q}_i+\left(\tilde{\mathcal{H}}-\mathcal{H}\right)$  para que la transformación $\left\{q_i, p_i\right\} \rightarrow\left\{Q_i, P_i\right\}$ sea canónica.
	\item<2-> Las derivadas parciales de  $\mathcal{F}$ con respecto a sus argumentos, contenidas en la expresión de $\frac{d  \mathcal{F}}{d t}$, permiten establecer la relación entre las variables $\left\{Q_i, P_i\right\}$ y $\left\{q_i, p_i\right\}$ y los tipos de función generadora.
	\item<3->  La transformación canónica asociada a una función generadora  $\mathcal{F}$ es una propiedad característica de la función  $\mathcal{F}$ y no depende del Hamiltoniano de un sistema específico. 
	\item<4-> Dada una función generadora $\mathcal{F}$, es posible encontrar una transformación canónica asociada a $\mathcal{F}$.
	\item<5-> Dada una transformación canónica, en principio es posible obtener la función generadora que produce esa transformación.
	\item<6-> La relación entre el Hamiltoniano $\mathcal{H}\left(q_i, p_i, t\right)$ y el Hamiltoniano transformado $\tilde{\mathcal{H}}\left(Q_i, P_i, t\right)$ resultante de la transformación canónica $\left\{q_i, p_i, t\right\} \rightarrow\left\{Q_i, P_i, t\right\}$ generada por una $\mathcal{F}$ siempre es $\tilde{\mathcal{H}}=\mathcal{H}+\frac{\partial \mathcal{F}}{\partial t}$
\end{itemize}
}

\section{Ejemplo de Función Generadora}
\frame{
\frametitle{Ejemplo de Función Generadora 1/2}
Encontrar la transformación canónica $\left\{q_i, p_i\right\} \rightarrow\left\{Q_i, P_i\right\}, i=1,2$, generada por la función $\mathcal{G}=q_1\left(P_1+2 p_2\right)+p_2 P_2$.
\begin{itemize}  
	\item<1-> $\mathcal{G}\left(q_1, P_1, p_2, P_2\right) \Rightarrow $ 
	$\frac{d \mathcal{G}}{d t} =\frac{\partial \mathcal{G}}{\partial q_1} \dot{q}_1+\frac{\partial \mathcal{G}}{\partial P_1} \dot{P}_1+\frac{\partial \mathcal{G}}{\partial p_2} \dot{p}_2+\frac{\partial \mathcal{G}}{\partial P_2} \dot{P}_2 \Rightarrow $ \\ $ \frac{d \mathcal{G}}{d t} = \left(P_1+2 p_2\right) \dot{q}_1+q_1 \dot{P}_1+\left(2 q_1+P_2\right) \dot{p}_2+p_2 \dot{P}_2$
	\item<2-> Comparando con $\frac{d \mathcal{F}}{d t}  =\sum_{i=1}^2 p_i \dot{q}_i-\sum_{i=1}^2 P_i \dot{Q}_i+\left(\tilde{\mathcal{H}}-\mathcal{H}\right) \Rightarrow $ \\ $\frac{d \mathcal{F}}{d t}= p_1 \dot{q}_1+p_2 \dot{q}_2-P_1 \dot{Q}_1-P_2 \dot{Q}_2+\left(\tilde{\mathcal{H}}-\mathcal{H}\right)$
	\item<3-> Para llevarlo a esa forma hacemos $p_2 \dot{q}_2 =\frac{d}{d t}\left(p_2 q_2\right)-q_2 \dot{p}_2$; $\quad P_1 \dot{Q}_1 =\frac{d}{d t}\left(P_1 Q_1\right)-Q_1 \dot{P}_1$;  $\quad P_2 \dot{Q}_2 =\frac{d}{d t}\left(P_2 Q_2\right)-Q_2 \dot{P}_2$
	\item<4-> Con lo cual $\frac{d \mathcal{F}}{d t}=p_1 \dot{q}_1-q_2 \dot{p}_2+Q_1 \dot{P}_1+Q_2 \dot{P}_2+\frac{d}{d t}\left(p_2 q_2-P_1 Q_1-P_2 Q_2\right)+\left(\tilde{\mathcal{H}}-\mathcal{H}\right)$ 
\end{itemize}
}

\section{Ejemplo: $Q=q+e^p \quad \& \quad P=p$}
\frame{
\frametitle{Ejemplo: $Q=q+e^p \quad \& \quad P=p$}
Dada una transformación $Q=q+e^p \quad \& \quad P=p$, encontrar sus función generadora
\begin{itemize}  
	\item<1-> Para encontrar la función generadora $ F $ supondremos una función generadora del tipo $F_2(q, P)$, que depende de  $q$ y  $P$
	\item<2-> Las relaciones para \( F_2(q, P) \) son: $p = \frac{\partial F_2}{\partial q} \quad \text{y} \quad Q = \frac{\partial F_2}{\partial P}$
	\item<3-> Sustituyendo \( p = P \) en  $Q = q + e^P$, como \( Q = \frac{\partial F_2}{\partial P} \Rightarrow \), $ F_2(q, P) = qP + \int e^P \, dP \Rightarrow F_2(q, P) = qP + e^P + C$ y $\quad C= 0$
	\item<4->  La función generadora es $F_2(q, P) = qP + e^P$

\end{itemize}
}

\frame{
\frametitle{Título transparencia}
Las funciones generadores pueden no depender de todas las variables $\left(q_i, p_i, Q_i, P_i, t\right)$ y 
 $\frac{d  \mathcal{F}}{d t}=\sum_{i=1}^s p_i \dot{q}_i-\sum_{i=1}^s P_i \dot{Q}_i+\left(\tilde{\mathcal{H}}-\mathcal{H}\right)$ nos permite identificar varias 
\begin{itemize}
	\item<1-> La función $\mathcal{F}_1\left(q_i, Q_i, t\right)$ genera  $p_i=p_i(q, Q, t), P_i=P_i(q, Q, t)$  
	\item<1-> $\mathcal{F}_1=\mathcal{F}_1\left(q_i, Q_i, t\right) \Rightarrow \frac{d \mathcal{F}_1}{d t}=\sum_{i=1}^s\left(\frac{\partial \mathcal{F}_1}{\partial q_i} \dot{q}_i+\frac{\partial \mathcal{F}_1}{\partial Q_i} \dot{Q}_i\right)+\frac{\partial \mathcal{F}_1}{\partial t}$.
	\item<1-> $p_i=\frac{\partial \mathcal{F}_1}{\partial q_i} =p_i(q, Q, t); \quad P_i=-\frac{\partial \mathcal{F}_1}{\partial Q_i} =P_i(q, Q, t); \quad \tilde{\mathcal{H}} =\mathcal{H}+\frac{\partial \mathcal{F}_1}{\partial t}$
\end{itemize}
}
