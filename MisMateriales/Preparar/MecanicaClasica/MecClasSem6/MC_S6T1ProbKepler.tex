\section{El Campo gravitacional: problema de dos cuerpos (otra vez)}
\frame{
  \frametitle{El Campo gravitacional: problema de dos cuerpos}
   \begin{itemize}  
  	\item<1-> El módulo de la fuerza gravitacional entre dos partículas con masas $m_1$ y $m_2$, separadas por una distancia $r$ es $f(r)=-G \frac{m_1 m_2}{r^2}$, donde $G$ es la constante universal gravitacional.
	\item<2-> La fuerza gravitacional que una partícula de masa $m_1$ ejerce sobre otra partícula de masa $m_2$ es $ \mathbf{F}(\mathbf{r})=-\nabla V(r)=-G \frac{m_1 m_2}{r^2} \hat{\mathbf{r}}$, con $V(r)=-G \frac{m_1 m_2}{r}$
	\item<3-> Entonces la intensidad del campo gravitacional de $m_1$ en la posición $\mathbf{r}$ sobre la partícula $m_2$ es  $mathbf{g}(\mathbf{r}) \equiv \frac{\mathbf{F}(\mathbf{r})}{m_2}=-\frac{G m_1}{r^2} \hat{\mathbf{r}}$, donde definimos el potencial gravitacional $\varphi(\mathbf{r})$ producido por $m_1$ como  $\varphi(\mathbf{r}) \equiv \frac{V(r)}{m_2}=-\frac{G m_1}{r}$
    \end{itemize}
}

\section{Flujo del campo gravitacional}
\frame{
  \frametitle{Flujo del campo gravitacional}
   	\begin{itemize}
		\item<1-> Consideremos una partícula de masa $m$ dentro de una superficie arbitraria y cerrada $S$ que contiene un volumen V.
		\begin{figure}[t]
		\includegraphics[width=2.0in]{Figuras/FlujoGravitacional.png}
   		\end{figure}  
  	\item<2-> El flujo del campo gravitacional a través de la superficie $S$ es $\Phi=\oint_S \mathbf{g} \cdot d \mathbf{A}  =-G m \oint_S \frac{1}{r^2} d A \; \hat{\mathbf{r}} \cdot \hat{\mathbf{n}}$ donde $g$ se evalúa sobre $S$ y $d \mathbf{A}=d A \;\hat{\mathbf{n}}$ es el diferencial de área de $S$ con un vector normal unitario $\hat{\mathbf{n}}$.
	\item<3-> Entonces, $\oint_S \mathbf{g} \cdot d \mathbf{A}=-G m \oint_S d \Omega=-4 \pi G m$ con $d \Omega=\frac{d A \cos \theta}{r^2}=\frac{d A \hat{\mathbf{n}} \cdot \hat{\mathbf{r}}}{r^2}$ el diferencial de ángulo sólido, con origen en $m$ encierra un área $d A \cos \theta$ a la distancia $r$
	\item<4-> El flujo del campo gravitacional a través de $S$ es proporcional a la masa de la partícula encerrada por $S$, independientemente de la ubicación de la partícula dentro de la superficie $S$.
    \end{itemize}
}

\section{La ecuación de Poisson}
\frame{
  \frametitle{La ecuación de Poisson}
   \begin{itemize}  
  	\item<1-> El flujo total para un sistema de partículas con masas $m_i, i=1, \ldots, N$, encerradas por la superficie $S$, es $ \oint_S \mathbf{g} \cdot d \mathbf{A}=-4 \pi G \sum_{i=1}^N m_i=-4 \pi G M_{\mathrm{enc}}$ donde $M_{\text {enc }}$ es la masa total encerrada por $S$.
	\item<2-> El teorema de la divergencia para el campo $\mathbf{g}$ nos dice $\oint_S \mathbf{g} \cdot d \mathbf{A}=\int_{\mathrm{V}} \nabla \cdot \mathbf{g} d \mathrm{~V}=-4 \pi G \int_{\mathrm{V}} \rho d \mathrm{~V}$ donde $M_{\mathrm{enc}}=\int_{\mathrm{V}} \rho d \mathrm{~V}$
	\item<3-> Puesto que el volumen V es arbitrario tendremos $\nabla \cdot \mathbf{g}=-4 \pi G \rho$ 
	\item<4-> Que se traduce en la ecuación de Poisson $\nabla^2 \varphi=4 \pi G \rho$
    \end{itemize}
}

\section{El problema inverso}
\frame{
  \frametitle{El problema inverso}
   \begin{itemize}  
  	\item<1-> Consideremos el problema inverso: dada una órbita $r(\theta)$, determinar el potencial $V(r)$, o la fuerza central $f(r)$, que causa ésta órbita.
	\item<2-> La ecuación de movimiento para $r(t)$ es $\mu \ddot{r}-\frac{L^2}{\mu r^3}=-\frac{\partial V}{\partial r}$
	\item<3->  La derivada temporal de $r(t)$ se puede expresar, usando $\dot{\theta}=\frac{L}{\mu r^2}$, como $\dot{r}=\frac{d r}{d t}=\frac{d r}{d \theta} \frac{d \theta}{d t}=\frac{L}{\mu r^2} \frac{d r}{d \theta}$
	\item<4-> En general, la derivada temporal se puede escribir como $\frac{d}{d t}  =\frac{L}{\mu r^2} \frac{d}{d \theta}$
	\item<5-> Con lo cual $\ddot{r}  =\frac{d}{d t}\left(\frac{d r}{d t}\right)=\frac{d}{d t}\left(\frac{L}{\mu r^2} \frac{d r}{d \theta}\right)  =\frac{L}{\mu r^2} \frac{d}{d \theta}\left(\frac{L}{\mu r^2} \frac{d r}{d \theta}\right)$
	\item<6-> Es decir $\frac{l}{r^2} \frac{d}{d \theta}\left(\frac{l}{\mu r^2} \frac{d r}{d \theta}\right)-\frac{L^2}{\mu r^3}=-\frac{\partial V}{\partial r}$
	\item<7-> Si $u=1 / r$, tenemos  $\frac{L^2}{\mu} u^2 \frac{d}{d \theta}\left(\frac{d u}{d \theta}\right)+\frac{L^2}{\mu} u^3=-u^2 \frac{\partial V}{\partial u}$
	\item<8-> es decir, $ \frac{L^2}{\mu}\left[\frac{d^2 u}{d \theta^2}+u\right]=-\frac{\partial V}{\partial u}$
	\item<9-> Constituye la ecuación diferencial de la órbita para $r(\theta)=1 / u(\theta)$ en términos del potencial $V(r)=V(1 / u)$. Se conoce como la ecuación de Binet.
    \end{itemize}
}

\section{El problema de Kepler}
\frame{
  \frametitle{El problema de Kepler}
   \begin{itemize}  
  	\item<1-> El problema de Kepler se refiere al cálculo de la órbita de una partícula sobre la cual actúa fuerza gravitacional $f(r)=-\frac{k}{r^2}$, con $k=G m_1 m_2$.
	\item<2-> La órbita $r(\theta)$ correspondiente la fuerza gravitacional puede ser determinada a partir de la ecuación de Binet $\frac{L^2}{\mu}\left(\frac{d^2 u}{d \theta^2}+u\right)=-\frac{\partial V}{\partial u}$
	\item<3-> Como el potencial gravitacional es $V(r)=-\frac{k}{r}=-k u$, obtenemos $\frac{d^2 u}{d \theta^2}+u=k \frac{\mu}{L^2}$. Una ecuación diferencial ordinaria inhomogénea de segundo orden
	\item<4-> La solución de la ecuación homogénea es $u_h^{\prime \prime}+u_h=0 \Rightarrow u_h=A \cos \left(\theta-\theta_0\right)$ con  $A$ y  $\theta_0$ constantes
	\item<5-> Una solución particular de la inhomogénea es $u_p=k \frac{\mu}{L^2}$.
	\item<6-> la solución general $u(\theta)=u_h+u_p$ es 
	$u(\theta) =\frac{k \mu}{L^2}+A \cos \left(\theta-\theta_0\right) \Rightarrow \frac{1}{r} =\frac{k \mu}{L^2}\left[1+e \cos \left(\theta-\theta_0\right)\right],$ y $e$  const.
    \end{itemize}
}

\section{Las cónicas}
\frame{
  \frametitle{Las cónicas}
   \begin{itemize}  
  	\item<1-> La forma general de la ecuación es una cónica con el origen en uno de los focos $\frac{q}{r}=1+e \cos \theta \Leftrightarrow \frac{1}{r}=\frac{\mu k}{L^2}\left(1+\sqrt{1+\frac{2 E L^2}{\mu k^2}} \cos \theta\right)$, \\para  $\theta_0=0$ para $t=0$
	\item<2-> El movimiento de la partícula con masa reducida $\mu$ en el potencial $V=-k / r$ sigue la trayectoria de una sección cónica 
	\item<3- >El tipo de cónica (circunferencia, elipse, parábola o hipérbola) depende del valor de la excentricidad $e$, i.e. de la energía total $E$.
	\item<4-> La excentricidad de la órbita es $e=\sqrt{1+\frac{2 E L^2}{\mu k^2}}$
	\item<5-> El latus $q$ es  $r\left(\frac{\pi}{2}\right)\equiv q =\frac{L^2}{\mu k}$ y el $r_{\min }=r(0)=\frac{q}{1+e}$
	\begin{figure}[t]
	\includegraphics[width=2.0in]{Figuras/Conica.png}
   	\end{figure}   
    \end{itemize}
