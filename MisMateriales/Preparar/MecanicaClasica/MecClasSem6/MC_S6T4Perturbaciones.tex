\section{Estabilidad de órbitas circulares}
\frame{
  \frametitle{Estabilidad de órbitas circulares}
  \begin{itemize}  
  	\item<1-> Sea radio $r_0$ de una órbita circular descrita por una partícula sujeta a un potencial efectivo $V_{\mathrm{ef}}(r)=V(r)+\frac{L^2}{2 \mu r^2}$
	\item<2-> Entonces, un mínimo es  $\left.\frac{\partial V_{\mathrm{ef}}}{\partial r}\right|_{r_0}  = 0 \left.\Rightarrow \frac{\partial V}{\partial r}\right|_{r_0}-\frac{L^2}{\mu r_0^3} = 0$ 
	\item<3-> y un verdadero mínimo cumple $\left.\frac{\partial^2 V_{\mathrm{ef}}}{\partial r^2}\right|_{r_0}  > 0 \left.\Rightarrow \frac{\partial^2 V}{\partial r^2}\right|_{r_0}+\frac{3 L^2}{\mu r_0^4}  >0$
	\item<4-> La fuerza central atractiva sobre una partícula en la órbita circular de radio $r_0$ es $f\left(r_0\right)=-\left.\frac{\partial V}{\partial r}\right|_{r_0}=-\frac{L^2}{\mu r_0^3}$
	\item<5-> con lo cual $-\left.\frac{\partial f}{\partial r}\right|_{r_0}-\frac{3 f\left(r_0\right)}{r_0}>0 \Rightarrow  \frac{3 f\left(r_0\right)}{r_0}+f^{\prime}\left(r_0\right)<0$
	\item<6-> En general para fuerzas de la forma $f(r)=-k r^n(k>0)$, la condición de estabilidad se cumple para $n>-3$
	\item<7-> La fuerza gravitacional $(n=-2)$ y la fuerza de un resorte $(n=1)$ producen órbitas circulares estables.

   \end{itemize}
}

\section{Perturbaciones $r=r_0+\eta$}
\frame{
  \frametitle{Perturbaciones $r=r_0+\eta$}
   \begin{itemize}  
  	\item<1-> Supongamos que la energía de la partícula es $E>V_{\mathrm{ef}}\left(r_0\right)$, donde $r_0$ es el radio de la órbita circular estable. 
	\item<2-> Consideremos una oscilación radial de pequeña amplitud $\eta$ alrededor del radio de la órbita circular $r_0$. Esto es $r=r_0+\eta$, con $\eta / r_0 \ll 1$
	\begin{figure}[t]
		\includegraphics[width=2.0in]{Figuras/EstabilidadEta.png}
   	\end{figure}
	\item<3-> Desarrollamos por Taylor de la función $V_{\mathrm{ef}}(r)$ alrededor de $r_0$, y tenemos
	$V_{\mathrm{ef}}(r)=V_{\mathrm{ef}}\left(r_0\right)+ \cancelto{0}{\left.\frac{\partial V_{\mathrm{ef}}}{\partial r}\right|_{r_0}}\left(r-r_0\right)+\left.\frac{1}{2} \frac{\partial^2 V_{\mathrm{ef}}}{\partial r^2}\right|_{r_0}\left(r-r_0\right)^2+\ldots$
	\item<4-> El primer término es una constante, $V_{\mathrm{ef}}\left(r_0\right)=$ cte, y puede ser suprimido  el segundo término se anula por ser un mínimo.

    \end{itemize}

\section{Precesión}
\frame{
  \frametitle{Precesión}
   \begin{itemize}  
  	\item<1-> El ángulo de precesión $\Delta \theta$ es el recorrido por la dirección del perihelio en el plano del movimiento durante un período de oscilación radial $T_r$. 
	\item<2-> Esto es $\Delta \theta=\dot{\theta} T_r=2 \pi \frac{\dot{\theta}}{\omega_r}$, y como  $\dot{\theta}=\frac{L}{\mu r_0^2}$, 
	\item<3-> tenemos $\Delta \theta=2 \pi \frac{L}{\mu r_0^2}\left(\left.\frac{1}{\mu} \frac{\partial^2 V_{\mathrm{ef}}}{\partial r^2}\right|_{r_0}\right)^{-1 / 2}$
	\item<4-> La dirección del perihelio $r_{\min }$ cambia en un ángulo $\Delta \theta$ durante la precesión de la órbita. 
	\item<5-> La condición para que ocurra precesión es que $\Delta \theta<2 \pi$; es decir, $\dot{\theta}<\omega_r$.

    \end{itemize}
